@app.get("/corpus/search")
async def corpus_search(q: str = "", subject_type: str = "", min_hurst: float = 0.0,
                        flagged_only: bool = False, medium: str = ""):
    clauses, params = ["1=1"], []
    if subject_type:
        clauses.append("subject_type=?"); params.append(subject_type)
    if min_hurst > 0:
        clauses.append("hurst>=?"); params.append(min_hurst)
    if flagged_only:
        clauses.append("pars_tier2_flag=1")
    if medium:
        clauses.append("medium=?"); params.append(medium)
    sql = f"SELECT * FROM corpus WHERE {' AND '.join(clauses)} ORDER BY indexed_at DESC LIMIT 50"
    with get_db() as conn:
        rows = conn.execute(sql, params).fetchall()
    return {"results": [dict(r) for r in rows]}

_yolo_model = None

def _get_yolo():
    global _yolo_model
    if _yolo_model is None:
        from ultralytics import YOLO
        _yolo_model = YOLO("yolov8n.pt")
    return _yolo_model

def extract_frames(path: str) -> list:
    """Extract frames from video at target FPS."""
    try:
        import cv2
        cap = cv2.VideoCapture(path)
        src_fps = cap.get(cv2.CAP_PROP_FPS) or FPS_TARGET
        step = max(1, round(src_fps / FPS_TARGET))
        frames, fi = [], 0
        while True:
            ret, frame = cap.read()
            if not ret:
                break
            if fi % step == 0:
                frames.append(cv2.resize(frame, (THRU_W, THRU_H)))
            if len(frames) >= FPS_TARGET * MAX_DURATION:
                break
            fi += 1
        cap.release()
        if not frames:
            raise RuntimeError("No frames extracted. Check video file.")
        return frames
    except ImportError:
        raise RuntimeError("OpenCV not installed. Run: pip install opencv-python")

def detect_objects(frame, fi: int, total: int) -> list:
    """
    Detect objects in a single frame using YOLOv8.
    PRODUCTION INTEGRATION POINTS:
      - Grounding DINO → open-vocabulary detection
      - DepthAnything V2 → monocular depth
      - Farneback optical flow → velocity vectors
      - ByteTrack → persistent object identity
      - EasyOCR/PaddleOCR → text in frame
      - CLIP → attribute classification
    """
    try:
        model = _get_yolo()
        results = model(frame, verbose=False)[0]
        tags = []
        for box in results.boxes:
            x1, y1, x2, y2 = box.xyxy[0].tolist()
            conf = float(box.conf[0])
            label = results.names[int(box.cls[0])]
            x = x1 / THRU_W; y = y1 / THRU_H
            w = (x2 - x1) / THRU_W; h = (y2 - y1) / THRU_H
            cx, cy = x + w / 2, y + h / 2
            row = "T" if cy < 0.33 else "M" if cy < 0.67 else "B"
            col = "L" if cx < 0.33 else "C" if cx < 0.67 else "R"
            zone = row + col
            tags.append(Tag(
                object_id=f"{label}_{fi}_{len(tags)}", class_label=label,
                frame_index=fi, timestamp_ms=fi * (1000 / FPS_TARGET),
                bbox=[x, y, w, h], depth_score=0.5, orientation="unknown",
                screen_zone=zone, colors=["unknown"], action="unknown",
                action_confidence=0.5, first_seen_frame=fi,
                velocity=[0.0, 0.0], acceleration=[0.0, 0.0], state_changes=[],
                confidence=conf, is_high_confidence=(conf >= 0.85),
                medium_source=MEDIUM_VIDEO
            ))
        return tags
    except ImportError:
        raise RuntimeError("Detection models not installed. Run: pip install ultralytics")

def convert_video(path: str) -> Tuple[list, list]:
    """Full video conversion: extract frames → detect → return (frame_tags, frames)."""
    frames = extract_frames(path)
    N = len(frames)
    frame_tags = []
    for fi, frame in enumerate(frames):
        tags = detect_objects(frame, fi, N)
        frame_tags.append(tags)
    return frame_tags, frames

# ─── AUDIO CONVERTER ─────────────────────────────────────────────────────────
def convert_audio(path: str) -> Tuple[list, list]:
    """
    Convert audio to semantic tags via spectral analysis.
    Each 'frame' is a time window (hop length) of audio.
    Tags represent detected audio events, speech segments, spectral features.
    INTEGRATION: librosa, Whisper, YAMNet
    """
    try:
        import librosa
        y, sr = librosa.load(path, sr=22050, duration=MAX_DURATION)
    except ImportError:
        # Fallback: generate semantic structure from file metadata
        y = None; sr = 22050
    except Exception:
        y = None; sr = 22050

    hop = 512
    if y is not None:
        # Spectral analysis
        S = np.abs(librosa.stft(y, hop_length=hop))
        mel = librosa.feature.melspectrogram(y=y, sr=sr, hop_length=hop)
        n_frames = S.shape[1]
        duration_s = len(y) / sr
        # Onset detection for event boundaries
        onset_env = librosa.onset.onset_strength(y=y, sr=sr, hop_length=hop)
        onsets = librosa.onset.onset_detect(y=y, sr=sr, hop_length=hop, units='frames')
        # Chroma for harmonic content
        chroma = librosa.feature.chroma_stft(y=y, sr=sr, hop_length=hop)
        # RMS energy
        rms = librosa.feature.rms(y=y, hop_length=hop)[0]
        # Spectral centroid
        sc = librosa.feature.spectral_centroid(y=y, sr=sr, hop_length=hop)[0]
    else:
        n_frames = int(MAX_DURATION * sr / hop)
        S = np.zeros((1025, max(1, n_frames)))
        mel = np.zeros((128, max(1, n_frames)))
        onset_env = np.zeros(max(1, n_frames))
        onsets = []
        chroma = np.zeros((12, max(1, n_frames)))
        rms = np.zeros(max(1, n_frames))
        sc = np.zeros(max(1, n_frames))
        duration_s = 0

    # Map audio frames to target FPS
    n_target = min(n_frames, FPS_TARGET * MAX_DURATION)
    step = max(1, n_frames // n_target) if n_frames > 0 else 1

    frame_tags = []
    onset_set = set(onsets)
    for fi in range(n_target):
        src_fi = fi * step
        if src_fi >= n_frames:
            break
        tags = []
        timestamp_ms = fi * (1000 / FPS_TARGET)

        # Energy-based "object" detection
        energy = float(rms[min(src_fi, len(rms) - 1)]) if len(rms) > 0 else 0
        centroid = float(sc[min(src_fi, len(sc) - 1)]) if len(sc) > 0 else 0
        is_onset = src_fi in onset_set

        # Dominant chroma (pitch class)
        chroma_vec = chroma[:, min(src_fi, chroma.shape[1] - 1)] if chroma.shape[1] > 0 else np.zeros(12)
        dom_pitch = int(np.argmax(chroma_vec))
        pitch_names = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'A#', 'B']

        if energy > 0.01:
            # Map spectral centroid to spatial position (frequency → x, energy → y)
            cx = min(0.95, max(0.05, centroid / 8000.0))
            cy = min(0.95, max(0.05, 1.0 - energy * 10))
            zone_row = "T" if cy < 0.33 else "M" if cy < 0.67 else "B"
            zone_col = "L" if cx < 0.33 else "C" if cx < 0.67 else "R"

            label = "onset" if is_onset else f"tone_{pitch_names[dom_pitch]}"
            conf = min(0.99, energy * 5 + 0.3)

            tags.append(Tag(
                object_id=f"audio_{fi}_{0}", class_label=label,
                frame_index=fi, timestamp_ms=timestamp_ms,
                bbox=[cx - 0.05, cy - 0.05, 0.1, 0.1],
                depth_score=energy, orientation="omnidirectional",
                screen_zone=zone_row + zone_col,
                colors=[pitch_names[dom_pitch]], action="sounding",
                action_confidence=conf, first_seen_frame=fi,
                velocity=[0.0, 0.0], acceleration=[0.0, 0.0],
                state_changes=[], confidence=conf,
                is_high_confidence=(conf >= 0.85),
                medium_source=MEDIUM_AUDIO
            ))

            # Add harmonic content as secondary tags
            for p_idx in range(12):
                pv = float(chroma_vec[p_idx])
                if pv > 0.3 and p_idx != dom_pitch:
                    hx = (p_idx / 12.0) * 0.8 + 0.1
                    hy = 1.0 - pv
                    hr = "T" if hy < 0.33 else "M" if hy < 0.67 else "B"
                    hc = "L" if hx < 0.33 else "C" if hx < 0.67 else "R"
                    tags.append(Tag(
                        object_id=f"harmonic_{fi}_{p_idx}",
                        class_label=f"harmonic_{pitch_names[p_idx]}",
                        frame_index=fi, timestamp_ms=timestamp_ms,
                        bbox=[hx - 0.03, hy - 0.03, 0.06, 0.06],
                        depth_score=pv, orientation="harmonic",
                        screen_zone=hr + hc, colors=[pitch_names[p_idx]],
                        action="resonating", action_confidence=pv,
                        first_seen_frame=fi,
                        velocity=[0.0, 0.0], acceleration=[0.0, 0.0],
                        state_changes=[], confidence=pv,
                        is_high_confidence=(pv >= 0.85),
                        medium_source=MEDIUM_AUDIO
                    ))
        frame_tags.append(tags)

    return frame_tags, []  # no visual frames for audio

# ─── IMAGE CONVERTER ─────────────────────────────────────────────────────────
def convert_image(path: str) -> Tuple[list, list]:
    """
    Convert a single image to semantic tags.
    Creates a single-frame analysis (but duplicated to create a minimal
    temporal signal for the math engines to operate on).
    """
    try:
        import cv2
        img = cv2.imread(path)
        if img is None:
            raise RuntimeError(f"Cannot read image: {path}")
        img = cv2.resize(img, (THRU_W, THRU_H))
    except ImportError:
        raise RuntimeError("OpenCV not installed. Run: pip install opencv-python")

    # Detect objects in the single frame
    tags = detect_objects(img, 0, 1)
    for t in tags:
        t.medium_source = MEDIUM_IMAGE

    # Create a minimal temporal signal by repeating the frame with slight variations
    # This gives the math engines enough data points to work with
    n_synthetic_frames = max(30, FPS_TARGET * 2)  # 2 seconds worth
    frame_tags = []
    for fi in range(n_synthetic_frames):
        frame_copy = []
        for t in tags:
            tc = Tag(
                object_id=t.object_id, class_label=t.class_label,
                frame_index=fi, timestamp_ms=fi * (1000 / FPS_TARGET),
                bbox=t.bbox[:], depth_score=t.depth_score,
                orientation=t.orientation, screen_zone=t.screen_zone,
                colors=t.colors[:], action=t.action,
                action_confidence=t.action_confidence, first_seen_frame=0,
                velocity=[0.0, 0.0], acceleration=[0.0, 0.0],
                state_changes=[], confidence=t.confidence,
                is_high_confidence=t.is_high_confidence,
                medium_source=MEDIUM_IMAGE
            )
            frame_copy.append(tc)
        frame_tags.append(frame_copy)

    return frame_tags, [img] * n_synthetic_frames

# ─── LANGUAGE CONVERTER ──────────────────────────────────────────────────────
def convert_language(path: str) -> Tuple[list, list]:
    """
    Convert text/language/PDF into semantic tags on the topographic field.
    Each sentence/paragraph becomes a 'frame'. Entities, concepts, sentiment
    become 'objects' positioned in semantic space.
    
    Handles:
    - Plain text files (.txt, .md, .html, .csv)
    - PDF files (.pdf) — extracts text + renders pages as images
    - Any other text-ish file
    
    For PDFs with images (like illustrated books), we extract BOTH:
    1. Text → language tags (sentences, entities, concepts)
    2. Page images → YOLO detection tags (objects in illustrations)
    Then merge both tag streams into the same semantic field.
    """
    ext = Path(path).suffix.lower()
    text = ""
    page_images = []  # for PDFs with illustrations
    
    # ─── PDF HANDLING ─────────────────────────────────────────────────────
    if ext == ".pdf":
        text, page_images = _extract_pdf(path)
    else:
        # Plain text / markdown / html / csv
        try:
            with open(path, 'r', encoding='utf-8', errors='replace') as f:
                text = f.read()
        except Exception:
            # Try binary read and extract printable text
            try:
                with open(path, 'rb') as f:
                    raw = f.read()
                text = raw.decode('utf-8', errors='replace')
                # Strip non-printable chars
                text = re.sub(r'[^\x20-\x7E\n\r\t]', ' ', text)
                text = re.sub(r'\s+', ' ', text)
            except Exception:
                text = ""
    
    if not text.strip() and not page_images:
        return [[]], []
    
    # ─── TEXT → TAGS ──────────────────────────────────────────────────────
    sentences = re.split(r'(?<=[.!?])\s+', text)
    sentences = [s.strip() for s in sentences if len(s.strip()) > 5]
    if not sentences and text.strip():
        # Split by newlines or chunks if no sentence boundaries
        sentences = [chunk.strip() for chunk in text.split('\n') if len(chunk.strip()) > 5]
    if not sentences:
        sentences = [text[:500]] if text.strip() else ["[no text extracted]"]
    
    # Entity extraction via regex
    entity_patterns = {
        'person': r'\b[A-Z][a-z]+\s+[A-Z][a-z]+\b',
        'number': r'\b\d+\.?\d*\b',
        'concept': r'\b[A-Z][A-Z]+\b',
        'date': r'\b\d{4}[-/]\d{1,2}[-/]\d{1,2}\b',
        'location': r'\b(?:Street|Avenue|Road|City|Park|Island|Mountain|Lake|River|North|South|East|West)\b',
    }
    
    positive_words = {'good', 'great', 'excellent', 'positive', 'success', 'love',
                      'happy', 'best', 'win', 'growth', 'strong', 'improve', 'value',
                      'treasure', 'gold', 'jewel', 'beauty', 'wonder', 'secret', 'find',
                      'discover', 'hidden', 'precious', 'rich', 'fortune'}
    negative_words = {'bad', 'poor', 'fail', 'negative', 'loss', 'hate', 'worst',
                      'weak', 'decline', 'risk', 'danger', 'threat', 'harm',
                      'lost', 'buried', 'dark', 'shadow', 'death', 'fear'}
    
    frame_tags = []
    n_frames = min(len(sentences), FPS_TARGET * MAX_DURATION)
    
    for fi, sentence in enumerate(sentences[:n_frames]):
        tags = []
        timestamp_ms = fi * (1000 / FPS_TARGET)
        words = sentence.lower().split()
        word_count = len(words)
        
        pos_count = sum(1 for w in words if w.strip('.,!?;:') in positive_words)
        neg_count = sum(1 for w in words if w.strip('.,!?;:') in negative_words)
        sentiment = (pos_count - neg_count) / max(word_count, 1)
        complexity = len(set(words)) / max(word_count, 1)
        
        cx = max(0.05, min(0.95, 0.5 + sentiment * 0.3))
        cy = max(0.05, min(0.95, 0.5 - complexity * 0.3))
        zone_r = "T" if cy < 0.33 else "M" if cy < 0.67 else "B"
        zone_c = "L" if cx < 0.33 else "C" if cx < 0.67 else "R"
        
        conf = min(0.99, 0.5 + complexity * 0.3 + abs(sentiment) * 0.2)
        tags.append(Tag(
            object_id=f"sentence_{fi}", class_label="sentence",
            frame_index=fi, timestamp_ms=timestamp_ms,
            bbox=[cx - 0.1, cy - 0.05, 0.2, 0.1],
            depth_score=complexity, orientation="forward",
            screen_zone=zone_r + zone_c, colors=["text"],
            action="stating" if sentiment >= 0 else "questioning",
            action_confidence=conf, first_seen_frame=fi,
            velocity=[0.0, 0.0], acceleration=[0.0, 0.0],
            state_changes=[], confidence=conf,
            is_high_confidence=(conf >= 0.85),
            medium_source=MEDIUM_LANGUAGE,
            ocr_text=sentence[:200]
        ))
        
        # Extract entities
        for etype, pattern in entity_patterns.items():
            matches = re.findall(pattern, sentence)
            for mi, match in enumerate(matches[:5]):
                eh = hash(match) % 10000
                ex = (eh % 100) / 100.0 * 0.8 + 0.1
                ey = ((eh // 100) % 100) / 100.0 * 0.8 + 0.1
                er = "T" if ey < 0.33 else "M" if ey < 0.67 else "B"
                ec = "L" if ex < 0.33 else "C" if ex < 0.67 else "R"
                econf = min(0.95, 0.6 + len(match) * 0.02)
                tags.append(Tag(
                    object_id=f"entity_{etype}_{eh}",
                    class_label=f"entity_{etype}",
                    frame_index=fi, timestamp_ms=timestamp_ms,
                    bbox=[ex - 0.03, ey - 0.03, 0.06, 0.06],
                    depth_score=0.7, orientation="entity",
                    screen_zone=er + ec, colors=[etype],
                    action="identified", action_confidence=econf,
                    first_seen_frame=fi,
                    velocity=[0.0, 0.0], acceleration=[0.0, 0.0],
                    state_changes=[], confidence=econf,
                    is_high_confidence=(econf >= 0.85),
                    medium_source=MEDIUM_LANGUAGE,
                    ocr_text=match
                ))
        frame_tags.append(tags)
    
    # ─── PDF IMAGE PAGES → YOLO TAGS (merged into same field) ─────────
    if page_images:
        for pi, img in enumerate(page_images):
            # Map page images to frame indices after text frames
            fi = len(frame_tags)
            try:
                img_tags = detect_objects(img, fi, len(page_images))
                for t in img_tags:
                    t.medium_source = "image_in_pdf"
                frame_tags.append(img_tags)
            except Exception:
                frame_tags.append([])  # empty frame if detection fails
    
    # Ensure minimum frames for math engines
    while len(frame_tags) < 30:
        frame_tags.append([])
    
    return frame_tags, page_images


def _extract_pdf(path: str) -> Tuple[str, list]:
    """
    Extract text and page images from a PDF.
    Tries multiple methods in order of quality.
    """
    text = ""
    page_images = []
    
    # Method 1: Try PyPDF2 (lightweight, text only)
    try:
        import PyPDF2
        with open(path, 'rb') as f:
            reader = PyPDF2.PdfReader(f)
            pages_text = []
            for page in reader.pages:
                pt = page.extract_text()
                if pt:
                    pages_text.append(pt)
            text = "\n\n".join(pages_text)
        if text.strip():
            print(f"PDF text extracted via PyPDF2: {len(text)} chars from {len(reader.pages)} pages")
    except ImportError:
        pass
    except Exception as e:
        print(f"PyPDF2 failed: {e}")
    
    # Method 2: Try pdfplumber (better text extraction)
    if not text.strip():
        try:
            import pdfplumber
            with pdfplumber.open(path) as pdf:
                pages_text = []
                for page in pdf.pages:
                    pt = page.extract_text()
                    if pt:
                        pages_text.append(pt)
                text = "\n\n".join(pages_text)
            if text.strip():
                print(f"PDF text extracted via pdfplumber: {len(text)} chars")
        except ImportError:
            pass
        except Exception as e:
            print(f"pdfplumber failed: {e}")
    
    # Method 3: Try PyMuPDF/fitz (text + images)
    if not text.strip():
        try:
            import fitz  # PyMuPDF
            doc = fitz.open(path)
            pages_text = []
            for page_num in range(len(doc)):
                page = doc[page_num]
                pt = page.get_text()
                if pt:
                    pages_text.append(pt)
                # Extract page as image for YOLO
                if page_num < 20:  # limit to first 20 pages
                    pix = page.get_pixmap(matrix=fitz.Matrix(2, 2))
                    img_data = np.frombuffer(pix.samples, dtype=np.uint8).reshape(
                        pix.height, pix.width, pix.n)
                    if pix.n == 4:  # RGBA
                        img_data = img_data[:, :, :3]  # drop alpha
                    import cv2
                    img_resized = cv2.resize(img_data, (THRU_W, THRU_H))
                    page_images.append(img_resized)
            text = "\n\n".join(pages_text)
            doc.close()
            if text.strip():
                print(f"PDF extracted via PyMuPDF: {len(text)} chars, {len(page_images)} page images")
        except ImportError:
            pass
        except Exception as e:
            print(f"PyMuPDF failed: {e}")
    
    # Method 4: Raw binary text extraction (always works, lower quality)
    if not text.strip():
        try:
            with open(path, 'rb') as f:
                raw = f.read()
            # Extract readable text from PDF binary
            # PDF text is often between BT and ET markers, or in parentheses
            # Simple approach: grab all printable ASCII runs
            printable_runs = re.findall(rb'[\x20-\x7E]{10,}', raw)
            extracted = []
            for run in printable_runs:
                decoded = run.decode('ascii', errors='replace')
                # Filter out PDF commands and metadata
                if not any(cmd in decoded for cmd in ['/Type', '/Font', '/Page', 
                          'stream', 'endstream', 'endobj', '/Length',
                          '/Filter', '/Subtype', '/MediaBox', 'xref',
                          '%%EOF', '/Producer', '/Creator']):
                    # Clean up common PDF text encoding
                    decoded = decoded.replace('\\(', '(').replace('\\)', ')')
                    if len(decoded) > 15:  # skip short fragments
                        extracted.append(decoded)
            text = "\n".join(extracted)
            if text.strip():
                print(f"PDF raw extraction: {len(text)} chars from binary")
        except Exception as e:
            print(f"Raw PDF extraction failed: {e}")
    
    # Method 5: Render pages as images using pdf2image (requires poppler)
    if not page_images:
        try:
            from pdf2image import convert_from_path
            images = convert_from_path(path, first_page=1, last_page=20, dpi=150)
            import cv2
            for img_pil in images:
                img_np = np.array(img_pil)
                if len(img_np.shape) == 3 and img_np.shape[2] == 4:
                    img_np = img_np[:, :, :3]
                img_bgr = img_np[:, :, ::-1]  # RGB → BGR for OpenCV
                img_resized = cv2.resize(img_bgr, (THRU_W, THRU_H))
                page_images.append(img_resized)
            print(f"PDF pages rendered as images: {len(page_images)} pages")
        except ImportError:
            pass
        except Exception as e:
            print(f"pdf2image failed: {e}")
    
    return text, page_images

def build_frame_meta(frame, tags: list, fi: int, medium: str = "video") -> FrameMeta:
    n = len(tags)
    d = min(0.99, max(0.02, sum(t.confidence for t in tags) * 0.15))
    zc = {z: 0.0 for z in ("TL", "TC", "TR", "ML", "MC", "MR", "BL", "BC", "BR")}
    for t in tags:
        zc[t.screen_zone] = zc.get(t.screen_zone, 0) + t.confidence
    ts = sum(zc.values()) or 1.0
    spatial = [zc[z] / ts for z in ("TL", "TC", "TR", "ML", "MC", "MR", "BL", "BC", "BR")]
    return FrameMeta(
        frame_index=fi, timestamp_ms=fi * (1000 / FPS_TARGET),
        setting="unknown", lighting="unknown", camera_motion="unknown",
        semantic_density=round(d, 6), spatial_density_field=spatial,
        dominant_action=tags[0].action if tags else "none", object_count=n,
        medium=medium)

def tracking_pass(frame_tags: list) -> list:
    id_map = {}; label_trackers = {}
    for fi, ftags in enumerate(frame_tags):
        used = set()
        for tag in ftags:
            cx = tag.bbox[0] + tag.bbox[2] / 2
            cy = tag.bbox[1] + tag.bbox[3] / 2
            best_id, best_d = None, 0.15
            for pcx, pcy, pid in label_trackers.get(tag.class_label, []):
                d = math.sqrt((cx - pcx) ** 2 + (cy - pcy) ** 2)
                if d < best_d and pid not in used:
                    best_d, best_id = d, pid
            if best_id is None:
                best_id = f"{tag.class_label}_{len(id_map)}"
            tag.object_id = best_id
            used.add(best_id)
            id_map.setdefault(best_id, []).append(tag)
        label_trackers = {}
        for tag in ftags:
            cx = tag.bbox[0] + tag.bbox[2] / 2
            cy = tag.bbox[1] + tag.bbox[3] / 2
            label_trackers.setdefault(tag.class_label, []).append(
                (cx, cy, tag.object_id))
    for oid, tags in id_map.items():
        for t in tags:
            t.duration_frames = len(tags)
            t.first_seen_frame = tags[0].frame_index
    return frame_tags

def relations_pass(frame_tags: list) -> list:
    zc = {"TL": (0.17, 0.17), "TC": (0.5, 0.17), "TR": (0.83, 0.17),
          "ML": (0.17, 0.5), "MC": (0.5, 0.5), "MR": (0.83, 0.5),
          "BL": (0.17, 0.83), "BC": (0.5, 0.83), "BR": (0.83, 0.83)}
    for fi, ftags in enumerate(frame_tags):
        for i, a in enumerate(ftags):
            ax, ay = zc.get(a.screen_zone, (0.5, 0.5))
            a.relations = []
            for j, b in enumerate(ftags):
                if i == j:
                    continue
                bx, by = zc.get(b.screen_zone, (0.5, 0.5))
                dx, dy = bx - ax, by - ay
                dist = math.sqrt(dx ** 2 + dy ** 2)
                angle = math.degrees(math.atan2(dy, dx)) % 360
                if dist < 0.15: pred = "adjacent_to"
                elif abs(dx) < 0.05: pred = "above" if dy < 0 else "below"
                elif abs(dy) < 0.05: pred = "left_of" if dx < 0 else "right_of"
                elif dy < 0: pred = "above_" + ("left_of" if dx < 0 else "right_of")
                else: pred = "below_" + ("left_of" if dx < 0 else "right_of")
                if a.class_label == "person" and b.class_label in ("hat", "glasses") and dist < 0.3:
                    pred = "wearing"
                if a.class_label == "person" and b.class_label in ("phone", "bag", "gun", "knife") and dist < 0.3:
                    pred = "holding"
                if a.class_label == "person" and b.class_label == "person" and dist < 0.2:
                    pred = "facing"
                a.relations.append(Relation(
                    subject_id=a.object_id, subject_label=a.class_label,
                    predicate=pred,
                    object_id=b.object_id, object_label=b.class_label,
                    distance_norm=round(dist, 4), angle_deg=round(angle, 2),
                    frame_index=fi, timestamp_ms=fi * (1000 / FPS_TARGET),
                    confidence=round(min(a.confidence, b.confidence), 4),
                ))
    return frame_tags

def temporal_pass(frame_tags: list) -> list:
    by_id = {}
    for ftags in frame_tags:
        for t in ftags:
            by_id.setdefault(t.object_id, []).append(t)
    for tags in by_id.values():
        for i in range(1, len(tags)):
            p, c = tags[i - 1], tags[i]
            c.velocity = [
                (c.bbox[0] + c.bbox[2] / 2) - (p.bbox[0] + p.bbox[2] / 2),
                (c.bbox[1] + c.bbox[3] / 2) - (p.bbox[1] + p.bbox[3] / 2)]
            c.acceleration = [c.velocity[0] - p.velocity[0],
                              c.velocity[1] - p.velocity[1]]
            for attr in ("action", "orientation", "screen_zone"):
                pv, cv2 = getattr(p, attr), getattr(c, attr)
                if pv != cv2:
                    c.state_changes.append(
                        {"frame": c.frame_index, "attr": attr, "from": pv, "to": cv2})
    # Anchor selection
    scores = {}
    for oid, tags in by_id.items():
        mc = sum(t.confidence for t in tags) / len(tags)
        dur = len(tags)
        nr = sum(len(t.relations) for t in tags) / max(len(tags), 1)
        scores[oid] = mc * math.log1p(dur) * math.log1p(nr + 1)
    if scores:
        anchor_id = max(scores, key=scores.get)
        for ftags in frame_tags:
            for t in ftags:
                t.centrality_score = scores.get(t.object_id, 0.0)
                t.is_anchor = (t.object_id == anchor_id)
    return frame_tags

def semantic_state_pass(frame_tags: list) -> list:
    """Build semantic state vectors for cosine angle computation."""
    zone_idx = {"TL": 0, "TC": 1, "TR": 2, "ML": 3, "MC": 4,
                "MR": 5, "BL": 6, "BC": 7, "BR": 8}
    action_idx = {"unknown": 0, "standing": 1, "walking": 2, "kneeling": 3,
                  "holding": 4, "gesturing": 5, "sitting": 6, "running": 7,
                  "facing": 8, "sounding": 9, "resonating": 10,
                  "stating": 11, "questioning": 12, "identified": 13}
    for ftags in frame_tags:
        for t in ftags:
            vec = [
                t.confidence, t.depth_score,
                t.bbox[0] + t.bbox[2] / 2, t.bbox[1] + t.bbox[3] / 2,
                t.bbox[2], t.bbox[3],
                t.velocity[0], t.velocity[1],
                float(zone_idx.get(t.screen_zone, 4)) / 8.0,
                float(action_idx.get(t.action, 0)) / 13.0,
                float(t.is_high_confidence),
            ]
            t.semantic_state_vec = vec
    # Compute semantic angles in relations
    by_id = {t.object_id: t for ftags in frame_tags for t in ftags}
    for ftags in frame_tags:
        for t in ftags:
            for rel in t.relations:
                ot = by_id.get(rel.object_id)
                if not ot or not t.semantic_state_vec or not ot.semantic_state_vec:
                    continue
                va = np.array(t.semantic_state_vec)
                vb = np.array(ot.semantic_state_vec)
                na, nb = np.linalg.norm(va), np.linalg.norm(vb)
                if na > 1e-6 and nb > 1e-6:
                    cos_sim = float(np.dot(va, vb) / (na * nb))
                    rel.semantic_angle = round(
                        math.degrees(math.acos(max(-1, min(1, cos_sim)))), 2)
    return frame_tags

def lexguard_pass(frame_tags: list, metas: list) -> list:
    """
    LexGuard PARS computation per frame.
    Full formulation from GP derivation:
      PARS = κ · Hz · Ex · Vu · (1 - Mt)
      Harmony = V − Burden − (α·PARS + β·Gap + γ·Fragility)
      SoCPM: REDIRECT when Harmony < 0

    GP-eligible: PARS budget is monomial ≤ constant.
    Log-convex reformulation preserves convexity.
    KKT multipliers become risk shadow prices.
    """
    alpha, beta, gamma, kappa = LEXGUARD_ALPHA, LEXGUARD_BETA, LEXGUARD_GAMMA, LEXGUARD_KAPPA
    for fi, (ftags, meta) in enumerate(zip(frame_tags, metas)):
        n = len(ftags)
        if n == 0:
            meta.pars_general = 0.0
            meta.pars_elevated = 0.0
            meta.harmony = 0.5
            continue
        # Hz: velocity-based behavioral intensity
        vels = [math.sqrt(t.velocity[0] ** 2 + t.velocity[1] ** 2) for t in ftags]
        mv = sum(vels) / len(vels) if vels else 0
        Hz = min(1.0, mv * 5 + meta.semantic_density * 0.3)
        # Ex: spatial concentration (exposure)
        central = sum(1 for t in ftags if t.screen_zone == "MC")
        Ex = min(1.0, central / max(n, 1) + meta.semantic_density * 0.2)
        # Vu: vulnerability context
        Vu = 0.5 + (0.15 if meta.setting in ("indoor", "vehicle") else 0)
        # Mt: mitigation factor (increases with available context)
        Mt = min(0.9, 0.1 + 0.05 * min(n, 10))
        # PARS Tier 1: general behavioral risk
        pg = min(1.0, kappa * Hz * Ex * Vu * (1 - Mt))
        # PARS Tier 2: elevated risk when flagged classes present
        flagged = any(t.class_label.lower() in FLAGGED_CLASSES for t in ftags)
        pe = min(1.0, kappa * min(1.0, Hz * 1.8) * Ex * Vu * (1 - Mt)) if flagged else 0.0
        # Gap: information gap (uncertainty)
        Gap = max(0, 1 - n / 5)
        # Fragility: behavioral fragility
        Fragility = 1 - mv  # static scenes are more fragile to perturbation
        # Harmony computation (full formulation)
        V = meta.semantic_density
        Burden = 0.1
        Safety_Tax = alpha * pg + beta * Gap + gamma * Fragility
        harmony = V - Burden - Safety_Tax
        # DRO adjustment: worst-case within Wasserstein ball
        # inf_{P: W(P,Phat)<=eps} E_P[V] >= E[V] - L*eps
        dro_penalty = LEXGUARD_DRO_EPSILON * abs(harmony) * 0.1
        harmony_robust = harmony - dro_penalty

        meta.pars_general = round(pg, 4)
        meta.pars_elevated = round(pe, 4)
        meta.harmony = round(harmony_robust, 4)

        # Per-tag PARS contribution
        for t in ftags:
            tv = math.sqrt(t.velocity[0] ** 2 + t.velocity[1] ** 2)
            t.pars_contribution = round(min(1.0, tv * kappa * 0.5), 4)
    return metas

class WaveEngine:
    """FFT, Morlet CWT (8 scales), SVD standing waves, semantic phonon spectrum."""
    def analyze(self, density, frame_tags, metas):
        D = np.array(density, dtype=np.float64)
        N = len(D); fps = float(FPS_TARGET)
        if N < 4:
            return self._empty()
        # FFT
        Np = 1
        while Np < N: Np *= 2
        Dp = np.pad(D, (0, Np - N))
        Df = np.fft.rfft(Dp)
        freqs = np.fft.rfftfreq(Np, d=1.0 / fps)
        power = np.abs(Df) ** 2
        phase = np.angle(Df)
        di = int(np.argmax(power[1:])) + 1
        domf = float(freqs[di])
        domp = 1.0 / max(domf, 1e-9)
        bwf = freqs[power >= power[di] / 2]
        bw = float(bwf[-1] - bwf[0]) if len(bwf) > 1 else 0.0
        p = power / (power.sum() + 1e-12)
        se = float(-np.sum(p[p > 0] * np.log2(p[p > 0])))
        sc2 = float(np.sum(freqs * power) / (power.sum() + 1e-12))
        # Morlet CWT
        scales = [2, 4, 8, 16, 32, 64, 128, 256]
        scalo = {}; scale_e = {}
        for s in scales:
            sig = s / (2 * math.pi)
            tw = np.arange(-3 * sig, 3 * sig + 1)
            pr = np.cos(2 * math.pi * tw / s) * np.exp(-tw ** 2 / (2 * sig ** 2))
            pi2 = np.sin(2 * math.pi * tw / s) * np.exp(-tw ** 2 / (2 * sig ** 2))
            c = np.convolve(D, pr, "same") + 1j * np.convolve(D, pi2, "same")
            mags = np.abs(c)
            scalo[f"s{s}"] = [round(float(v), 4) for v in mags[::5]]
            scale_e[f"s{s}"] = round(float(np.mean(mags ** 2)), 4)
        ds = max(scale_e, key=scale_e.get)
        # SVD standing waves
        if metas:
            Sm = np.array([m.spatial_density_field for m in metas], dtype=float).T
        else:
            Sm = np.zeros((9, N))
        U, Sig, Vt = np.linalg.svd(Sm, full_matrices=False)
        tv = float(np.sum(Sig ** 2))
        mv = [round(float(s ** 2 / (tv + 1e-12)), 4) for s in Sig]
        ns = int(np.sum(np.array(mv) > 0.05))
        # Object-level phase analysis
        obj_s = {}
        for fi2, ftags in enumerate(frame_tags):
            for t in ftags:
                if t.object_id not in obj_s:
                    obj_s[t.object_id] = np.zeros(N)
                if fi2 < N:
                    obj_s[t.object_id][fi2] = t.confidence
        oids = list(obj_s.keys())[:6]
        ievents = []; mpd = 0.0
        if len(oids) >= 2:
            dfs = []
            for i in range(len(oids) - 1):
                for j in range(i + 1, len(oids)):
                    sa, sb = obj_s[oids[i]], obj_s[oids[j]]
                    fa = np.fft.rfft(sa - sa.mean())
                    fb = np.fft.rfft(sb - sb.mean())
                    if di < len(fa) and di < len(fb):
                        ph = float(np.angle(fa[di] * np.conj(fb[di])))
                        dfs.append(abs(ph))
                    for k in range(0, N - 15, 8):
                        ca, cb = sa[k:k + 15], sb[k:k + 15]
                        if ca.std() > 1e-6 and cb.std() > 1e-6:
                            cc = float(np.corrcoef(ca, cb)[0, 1])
                            if cc > 0.75:
                                ievents.append({"frame": k, "type": "constructive",
                                                "objects": [oids[i], oids[j]],
                                                "r": round(cc, 3)})
            if dfs:
                mpd = float(np.mean(dfs))
        seen = set(); uev = []
        for ev in sorted(ievents, key=lambda x: -x["r"]):
            if ev["frame"] not in seen:
                uev.append(ev); seen.add(ev["frame"])
        it = "constructive" if mpd < math.pi / 4 else "destructive" if mpd > 3 * math.pi / 4 else "mixed"
        # Wave dynamics
        D2t = np.gradient(np.gradient(D))
        ls = float(np.mean(np.abs(np.gradient(np.gradient(Sm, axis=1), axis=1))) + 1e-12)
        wsc = float(math.sqrt(abs(float(np.abs(np.mean(D2t)) / ls))))
        Dsm = np.convolve(D, np.ones(5) / 5, "same")
        dg = np.gradient(Dsm)
        rf = [int(i) for i in range(1, N - 1)
              if dg[i - 1] > 0 > dg[i + 1] and D[i] > D.mean() + 0.7 * D.std()]
        # Phonons
        phonons = self._phonons(frame_tags, oids, obj_s, N)
        aoids = set(phonons.get("acoustic_objects", []))
        ooids = set(phonons.get("optical_objects", []))
        for ftags in frame_tags:
            for t in ftags:
                t.phonon_mode = ("acoustic" if t.object_id in aoids
                                 else "optical" if t.object_id in ooids
                                 else "none")
        return {
            "fourier": {
                "dominant_frequency_hz": round(domf, 5),
                "dominant_period_s": round(domp, 4),
                "dominant_phase_rad": round(float(phase[di]), 4),
                "bandwidth_hz": round(bw, 5),
                "spectral_entropy": round(se, 5),
                "spectral_centroid_hz": round(sc2, 5),
                "spectral_class": "narrowband" if bw < 0.05 else "broadband",
                "power_spectrum_sample": [round(float(v), 2) for v in power[1:min(80, len(power))]]
            },
            "wavelet": {"scale_energies": scale_e, "dominant_scale": ds,
                        "scalogram_sample": {k: v[:30] for k, v in scalo.items()}},
            "standing_waves": {"mode_variance": mv, "n_significant_modes": ns,
                               "dominant_spatial_pattern": [round(float(v), 4) for v in U[:, 0]]},
            "interference": {"mean_phase_diff_rad": round(mpd, 5), "type": it,
                             "events": uev[:6]},
            "wave_dynamics": {"wave_speed_c": round(wsc, 5),
                              "resonance_frames": rf[:6], "n_resonance": len(rf)},
            "phonons": phonons
        }

    def _phonons(self, frame_tags, oids, obj_s, N):
        all_ids = list({t.object_id for ft in frame_tags for t in ft})
        n = min(len(all_ids), 12)
        if n < 2:
            return {"acoustic_modes": [], "optical_modes": [],
                    "coupling_constant": 0.0, "acoustic_objects": [],
                    "optical_objects": [], "n_acoustic": 0, "n_optical": 0}
        idx = {oid: i for i, oid in enumerate(all_ids[:n])}
        A = np.zeros((n, n))
        for ftags in frame_tags:
            for t in ftags:
                i_ = idx.get(t.object_id)
                if i_ is None: continue
                for rel in t.relations:
                    j_ = idx.get(rel.object_id)
                    if j_ is None: continue
                    A[i_, j_] += rel.confidence / (rel.distance_norm + 0.1)
        A = (A + A.T) / 2
        A /= (A.max() + 1e-12)
        L = np.diag(A.sum(axis=1)) - A
        ev, evec = np.linalg.eigh(L)
        nz = ev > 1e-6; ev = ev[nz]; evec = evec[:, nz]
        if len(ev) == 0:
            return {"acoustic_modes": [], "optical_modes": [],
                    "coupling_constant": 0.0, "acoustic_objects": [],
                    "optical_objects": [], "n_acoustic": 0, "n_optical": 0}
        kappa = float(ev[0]); mid = float(np.median(ev))
        am = [round(float(e), 5) for e in ev[ev < mid]][:4]
        om = [round(float(e), 5) for e in ev[ev >= mid]][:4]
        ao = [all_ids[i] for i in range(n) if evec[i, 0] > 0][:5]
        oo = [all_ids[i] for i in range(n) if evec[i, -1] > 0][:5]
        return {"acoustic_modes": am, "optical_modes": om,
                "coupling_constant": round(kappa, 5),
                "acoustic_objects": ao, "optical_objects": oo,
                "n_acoustic": len(am), "n_optical": len(om)}

    def _empty(self):
        return {"fourier": {"dominant_frequency_hz": 0, "dominant_period_s": 0,
                            "dominant_phase_rad": 0, "bandwidth_hz": 0,
                            "spectral_entropy": 0, "spectral_centroid_hz": 0,
                            "spectral_class": "none", "power_spectrum_sample": []},
                "wavelet": {"scale_energies": {}, "dominant_scale": "none",
                            "scalogram_sample": {}},
                "standing_waves": {"mode_variance": [], "n_significant_modes": 0,
                                   "dominant_spatial_pattern": []},
                "interference": {"mean_phase_diff_rad": 0, "type": "none", "events": []},
                "wave_dynamics": {"wave_speed_c": 0, "resonance_frames": [], "n_resonance": 0},
                "phonons": {"acoustic_modes": [], "optical_modes": [],
                            "coupling_constant": 0, "acoustic_objects": [],
                            "optical_objects": [], "n_acoustic": 0, "n_optical": 0}}


class CalculusEngine:
    """D'(t), D''(t), ∫D(t)dt, ∇D, ∇²D, Hurst, Lyapunov, NLL anomaly scoring."""
    def analyze(self, density, frame_tags, metas):
        D = np.array(density, dtype=np.float64); fps = float(FPS_TARGET); N = len(D)
        if N < 4:
            return self._empty()
        D1 = np.gradient(D); D2 = np.gradient(D1); cum = np.cumsum(D) / fps
        tmass = float(np.trapezoid(D, dx=1 / fps))
        segs = [round(float(np.trapezoid(D[i:i + int(fps)], dx=1 / fps)), 5)
                for i in range(0, N, int(fps))]
        pf = int(np.argmax(np.abs(D2)))
        inf2 = [int(i) for i in range(1, N - 1) if np.sign(D1[i]) != np.sign(D1[i - 1])]
        m_, s_ = float(D.mean()), float(D.std() + 1e-12)
        sk_ = float(((D - m_) ** 3).mean() / (s_ ** 3))
        ku_ = float(((D - m_) ** 4).mean() / (s_ ** 4)) - 3.0
        H = self._hurst(D)
        ly = self._lyap(D)
        ents = []
        for ftags in frame_tags:
            cnt = Counter(t.class_label for t in ftags)
            tot = sum(cnt.values()) or 1
            ents.append(-sum((v / tot) * math.log2(v / tot + 1e-12) for v in cnt.values()))
        if metas:
            zD = np.array([m.spatial_density_field for m in metas], dtype=float)
            z3 = zD.reshape(N, 3, 3)
            gx2 = np.gradient(z3, axis=2).mean(axis=(1, 2))
            gy2 = np.gradient(z3, axis=1).mean(axis=(1, 2))
            gm = [round(float(math.sqrt(a ** 2 + b ** 2)), 4) for a, b in zip(gx2, gy2)]
            lap = (np.gradient(np.gradient(z3, axis=1), axis=1) +
                   np.gradient(np.gradient(z3, axis=2), axis=2)).mean(axis=(1, 2))
            ml = float(lap.mean())
        else:
            gm = [0.0] * N; ml = 0.0
        nll = [-0.5 * ((v - m_) / s_) ** 2 - 0.5 * math.log(2 * math.pi * s_ ** 2)
               for v in density]
        af = [int(i) for i, v in enumerate(nll) if v < float(np.percentile(nll, 5))]
        return {
            "density": {"mean": round(m_, 6), "std": round(s_, 6),
                        "min": round(float(D.min()), 6), "max": round(float(D.max()), 6),
                        "skewness": round(sk_, 6), "kurtosis": round(ku_, 6),
                        "coefficient_of_variation": round(s_ / (m_ + 1e-12), 6)},
            "derivatives": {"d1_series": [round(float(v), 4) for v in D1[::5]],
                            "d2_series": [round(float(v), 4) for v in D2[::5]],
                            "d1_mean_abs": round(float(np.mean(np.abs(D1))), 6),
                            "peak_impact_frame": pf,
                            "peak_impact_ts_s": round(pf / fps, 3)},
            "integrals": {"total_semantic_mass": round(tmass, 5),
                          "segment_masses_1s": segs,
                          "peak_segment_1s": int(np.argmax(segs)) if segs else 0,
                          "cumulative_sample": [round(float(v), 4) for v in cum[::5]]},
            "topology": {"inflection_frames": inf2[:20], "inflection_count": len(inf2),
                         "hurst_exponent": round(H, 6),
                         "hurst_interpretation": (
                             "persistent—long-range dependence" if H > 0.55
                             else "anti-persistent—mean-reverting" if H < 0.45
                             else "random walk"),
                         "lyapunov_exponent": round(ly, 6),
                         "lyapunov_interpretation": "chaotic" if ly > 0.05 else "stable"},
            "spatial": {"gradient_magnitude_series": gm[::5],
                        "mean_laplacian": round(ml, 6),
                        "peak_gradient_frame": int(np.argmax(gm)) if gm else 0},
            "information": {"mean_entropy_bits": round(float(np.mean(ents)) if ents else 0.0, 6),
                            "anomaly_frames": af[:10], "n_anomaly_frames": len(af)}
        }

    def _hurst(self, a):
        lags = [2, 4, 8, 16, 32, 64]; rv = []
        for lag in lags:
            segs = [a[i:i + lag] for i in range(0, len(a) - lag, lag)]; rs = []
            for seg in segs:
                if len(seg) < 2: continue
                m = seg.mean(); dev = np.cumsum(seg - m)
                R = dev.max() - dev.min(); S = seg.std() + 1e-12
                rs.append(R / S)
            if rs: rv.append(float(np.mean(rs)))
        if len(rv) < 2: return 0.5
        lx = np.log(np.array(lags[:len(rv)], float))
        ly = np.log(np.array(rv) + 1e-12)
        H, _ = np.polyfit(lx, ly, 1)
        return float(np.clip(H, 0, 1))

    def _lyap(self, D, eps=1e-5):
        divs = []
        for i in range(1, len(D) - 1):
            near = [j for j in range(max(0, i - 30), min(len(D), i + 30))
                    if 0 < abs(D[j] - D[i]) < eps * 100 and j != i]
            if not near: continue
            j = near[0]
            d0 = abs(D[i] - D[j]) + eps
            d1 = abs(D[min(i + 1, len(D) - 1)] - D[min(j + 1, len(D) - 1)]) + eps
            divs.append(math.log(d1 / d0))
        return float(np.mean(divs)) if divs else 0.0

    def _empty(self):
        return {"density": {"mean": 0, "std": 0, "min": 0, "max": 0,
                            "skewness": 0, "kurtosis": 0, "coefficient_of_variation": 0},
                "derivatives": {"d1_series": [], "d2_series": [], "d1_mean_abs": 0,
                                "peak_impact_frame": 0, "peak_impact_ts_s": 0},
                "integrals": {"total_semantic_mass": 0, "segment_masses_1s": [],
                              "peak_segment_1s": 0, "cumulative_sample": []},
                "topology": {"inflection_frames": [], "inflection_count": 0,
                             "hurst_exponent": 0.5, "hurst_interpretation": "insufficient data",
                             "lyapunov_exponent": 0, "lyapunov_interpretation": "stable"},
                "spatial": {"gradient_magnitude_series": [], "mean_laplacian": 0,
                            "peak_gradient_frame": 0},
                "information": {"mean_entropy_bits": 0, "anomaly_frames": [],
                                "n_anomaly_frames": 0}}


class GraphEngine:
    """Laplacian eigendecomposition, algebraic connectivity, betweenness, communities."""
    def analyze(self, frame_tags):
        fps = float(FPS_TARGET); pf = []; ch = []; aoids = set()
        for fi, ftags in enumerate(frame_tags):
            if not ftags:
                pf.append(None); continue
            G = nx.Graph()
            for t in ftags:
                aoids.add(t.object_id)
                G.add_node(t.object_id, label=t.class_label, conf=t.confidence)
            for t in ftags:
                for rel in t.relations:
                    w = rel.confidence / (rel.distance_norm + 0.1)
                    if G.has_edge(t.object_id, rel.object_id):
                        G[t.object_id][rel.object_id]["weight"] += w
                    else:
                        G.add_edge(t.object_id, rel.object_id, weight=w,
                                   predicate=rel.predicate)
            n = G.number_of_nodes()
            if n < 2:
                pf.append({"frame": fi, "nodes": n, "edges": 0, "alg_conn": 0.0})
                continue
            L = nx.laplacian_matrix(G, weight="weight").toarray().astype(float)
            ev = sorted(np.linalg.eigvalsh(L))
            ac = float(ev[1]) if len(ev) > 1 else 0.0
            try:
                bc = nx.betweenness_centrality(G, weight="weight")
                mc = max(bc, key=bc.get)
            except:
                mc = list(G.nodes)[0] if G.nodes else "none"
            ch.append(mc)
            try:
                comms = list(nx.community.greedy_modularity_communities(G, weight="weight"))
                nc = len(comms)
                mod = nx.community.modularity(G, comms, weight="weight")
            except:
                nc = 1; mod = 0.0
            pf.append({"frame": fi, "ts_s": round(fi / fps, 3), "nodes": n,
                        "edges": G.number_of_edges(),
                        "alg_conn": round(ac, 5),
                        "spectral_gap": round(float(ev[2] - ev[1]) if len(ev) > 2 else 0.0, 5),
                        "graph_density": round(nx.density(G), 5),
                        "most_central": mc, "n_communities": nc,
                        "modularity": round(mod, 5)})
        valid = [f for f in pf if f]
        ct = [i for i in range(1, len(ch)) if ch[i] != ch[i - 1]]
        dom = max(set(ch), key=ch.count) if ch else None
        return {

class CausalEngine:
    """Transfer entropy T(A→B), Granger VAR(1), before/after event tables."""
    def analyze(self, frame_tags):
        N = len(frame_tags); series = {}
        for fi, ftags in enumerate(frame_tags):
            for t in ftags:
                while len(series.get(t.object_id, [])) < fi:
                    series.setdefault(t.object_id, []).append(0.0)
                series.setdefault(t.object_id, []).append(t.confidence)
        for oid in series:
            while len(series[oid]) < N: series[oid].append(0.0)
        oids = list(series.keys())[:8]; granger = {}; te = {}
        for a in oids:
            for b in oids:
                if a == b: continue
                sa = np.array(series[a][:N]); sb = np.array(series[b][:N])
                if len(sa) < 6: continue
                sbt = sb[1:]; sbl = sb[:-1]; sal = sa[:-1]
                rr = float(np.mean((sbt - sbt.mean()) ** 2))
                X = np.column_stack([np.ones(len(sbl)), sbl, sal])
                try:
                    bt, _, _, _ = np.linalg.lstsq(X, sbt, rcond=None)
                    ru = float(np.mean((sbt - X @ bt) ** 2))
                    granger[f"{a}->{b}"] = round(
                        float(max(0, (rr - ru) / (ru / (len(sbt) - 3) + 1e-12))), 4)
                except: pass
                sad = (sa > sa.mean()).astype(int)
                sbd = (sb > sb.mean()).astype(int)
                if len(sbd) > 2:
                    bt2 = sbd[2:]; bl = sbd[1:-1]; al = sad[1:-1]
                    tval = self._ce(bt2, bl) - self._ce2(bt2, bl, al)
                    te[f"{a}->{b}"] = round(float(max(0, tval)), 5)
        st = {}
        for k, v in te.items():
            src = k.split("->")[0]; st[src] = st.get(src, 0) + v
        ds = max(st, key=st.get) if st else None
        mk = max(te, key=te.get) if te else None
        return {
            "granger": dict(sorted(granger.items(), key=lambda x: -x[1])[:10]),
            "transfer_entropy": dict(sorted(te.items(), key=lambda x: -x[1])[:10]),
            "before_after": self._ba(frame_tags),
            "summary": {"dominant_source": ds, "max_te_pair": mk,
                        "max_te_bits": te.get(mk, 0.0) if mk else 0.0,
                        "significant_pairs": len([v for v in te.values() if v > 0.02])}
        }

    def _ce(self, y, x):
        st = {}
        for i in range(len(y)): st.setdefault(int(x[i]), []).append(int(y[i]))
        tot = len(y); H = 0.0
        for vals in st.values():
            ps = len(vals) / tot; p1 = sum(vals) / len(vals); p0 = 1 - p1
            H += ps * (-(p1 * math.log2(p1 + 1e-12) + p0 * math.log2(p0 + 1e-12)))
        return H

    def _ce2(self, y, x1, x2):
        st = {}
        for i in range(len(y)):
            st.setdefault((int(x1[i]), int(x2[i])), []).append(int(y[i]))
        tot = len(y); H = 0.0
        for vals in st.values():
            ps = len(vals) / tot; p1 = sum(vals) / len(vals); p0 = 1 - p1
            H += ps * (-(p1 * math.log2(p1 + 1e-12) + p0 * math.log2(p0 + 1e-12)))
        return H

    def _ba(self, frame_tags):
        density = [sum(t.confidence for t in ftags) * 0.1 for ftags in frame_tags]
        D = np.array(density)
        events = [i for i in range(30, len(D) - 30)
                  if D[i] > D.mean() + 1.5 * D.std()]
        results = []
        for ev in events[:3]:
            before = Counter(); after = Counter()
            for fi in range(ev - 30, ev):
                if fi >= 0:
                    for t in frame_tags[fi]: before[t.class_label] += 1
            for fi in range(ev + 1, ev + 31):
                if fi < len(frame_tags):
                    for t in frame_tags[fi]: after[t.class_label] += 1
            results.append({"event_frame": ev, "event_ts_s": round(ev / FPS_TARGET, 2),
                            "before": dict(before.most_common(5)),
                            "after": dict(after.most_common(5)),
                            "appeared": list({k for k in after if k not in before}),
                            "disappeared": list({k for k in before if k not in after})})
        return results


class DynamicalEngine:
    """Takens embedding, correlation dimension, attractor classification, bifurcations."""
    def analyze(self, density):
        D = np.array(density, dtype=np.float64); fps = float(FPS_TARGET); N = len(D)
        if N < 20:
            return self._empty()
        dim = 3; lag = 5
        X = (np.array([D[i:i + (dim - 1) * lag + 1:lag] for i in range(N - (dim - 1) * lag)])
             if N > dim * lag + 10 else D.reshape(-1, 1))
        n = min(len(X), 200); Xs = X[:n]
        if n > 10:
            dists = cdist(Xs, Xs)
            ev2 = np.logspace(-3, 0, 20)
            Ce = [float(np.mean(dists < e)) for e in ev2]
            vl = [(math.log(e), math.log(c)) for e, c in zip(ev2, Ce) if c > 0]
            cd = float(np.polyfit([v[0] for v in vl], [v[1] for v in vl], 1)[0]) if len(vl) > 2 else 1.0
        else:
            cd = 1.0
        ll = []
        for i in range(0, N - 30, 15):
            seg = D[i:i + 30]
            diffs = np.abs(np.diff(seg))
            ll.append((i, float(np.mean(np.log(diffs + 1e-12)))))
        bif = []
        if len(ll) > 2:
            lv = np.array([v[1] for v in ll])
            ld = np.abs(np.diff(lv))
            thresh = lv.std() * 1.5
            bif = [ll[i + 1][0] for i, v in enumerate(ld) if v > thresh]
        lm = self._lyap(D)
        att = ("strange (chaotic)" if lm > 0.05 and cd > 1.5
               else "fixed point" if lm <= 0 and cd < 1.2
               else "quasi-periodic")
        thr = float(D.mean()); cros = []
        for i in range(1, N):
            if D[i - 1] < thr <= D[i]:
                cros.append({"frame": i, "type": "up", "value": round(float(D[i]), 4)})
            elif D[i - 1] > thr >= D[i]:
                cros.append({"frame": i, "type": "down", "value": round(float(D[i]), 4)})
        upc = [c["frame"] for c in cros if c["type"] == "up"]
        mp = float(np.mean(np.diff(upc))) if len(upc) > 1 else 0.0
        return {
            "state_space": {"embedding_dim": dim, "embedding_lag": lag,
                            "correlation_dimension": round(cd, 3),
                            "attractor_type": att},
            "stability": {"lyapunov_max": round(lm, 5),
                          "interpretation": "chaotic" if lm > 0.05 else "stable"},
            "bifurcations": {"n_bifurcations": len(bif), "frames": bif[:6],
                             "timestamps_s": [round(f / fps, 2) for f in bif[:6]]},
            "poincare": {"n_crossings": len(cros),
                         "mean_period_frames": round(mp, 2),
                         "mean_period_s": round(mp / fps, 3)}
        }

    def _lyap(self, D, eps=1e-5):
        divs = []
        for i in range(1, len(D) - 1):
            near = [j for j in range(max(0, i - 25), min(len(D), i + 25))
                    if 0 < abs(D[j] - D[i]) < eps * 100 and j != i]
            if not near: continue
            j = near[0]
            d0 = abs(D[i] - D[j]) + eps
            d1 = abs(D[min(i + 1, len(D) - 1)] - D[min(j + 1, len(D) - 1)]) + eps
            divs.append(math.log(d1 / d0))
        return float(np.mean(divs)) if divs else 0.0

    def _empty(self):
        return {"state_space": {"embedding_dim": 3, "embedding_lag": 5,
                                "correlation_dimension": 1.0, "attractor_type": "insufficient data"},
                "stability": {"lyapunov_max": 0, "interpretation": "stable"},
                "bifurcations": {"n_bifurcations": 0, "frames": [], "timestamps_s": []},
                "poincare": {"n_crossings": 0, "mean_period_frames": 0, "mean_period_s": 0}}


class TensorEngine:
    """Tucker-3/HOSVD on frames×objects×attributes tensor."""
    def analyze(self, frame_tags):
        N = len(frame_tags)
        oc = Counter(t.object_id for ftags in frame_tags for t in ftags)
        top = [k for k, _ in oc.most_common(8)]
        attrs = ["confidence", "depth_score", "vx", "vy", "ax", "ay"]
        if len(top) < 2:
            return {"note": "Insufficient tracked objects", "effective_dimensionality": 0}
        T3 = np.zeros((N, len(top), len(attrs)))
        oi = {oid: i for i, oid in enumerate(top)}
        for fi, ftags in enumerate(frame_tags):
            for t in ftags:
                idx = oi.get(t.object_id)
                if idx is None: continue
                T3[fi, idx, :] = [t.confidence, t.depth_score,
                                  t.velocity[0], t.velocity[1],
                                  t.acceleration[0], t.acceleration[1]]
        def unfold(T, n):
            s = T.shape
            o = [n] + [i for i in range(len(s)) if i != n]
            return np.transpose(T, o).reshape(s[n], -1)
        nc = [min(3, s) for s in T3.shape]; Um = []
        for n in range(len(T3.shape)):
            Tn = unfold(T3, n)
            U2, s2, Vt2 = np.linalg.svd(Tn, full_matrices=False)
            Um.append(U2[:, :nc[n]])
        fv = [round(float(np.var(Um[0][:, i])), 5) for i in range(nc[0])]
        tv = sum(fv) + 1e-12
        ne = sum(1 for v in fv if v / tv > 0.05)
        dol = {top[i]: round(float(Um[1][i, 0]), 4)
               for i in range(min(len(top), Um[1].shape[0]))}
        dal = {attrs[i]: round(float(Um[2][i, 0]), 4)
               for i in range(min(len(attrs), Um[2].shape[0]))}
        return {"shape": list(T3.shape), "n_components": nc,
                "frame_variance": fv,
                "dominant_object_loadings": dol,
                "dominant_attribute_loadings": dal,
                "effective_dimensionality": ne,
                "interpretation": (f"Rank-{ne}: structurally simple" if ne <= 2
                                   else f"Rank-{ne}: genuine multi-modal complexity")}

class SemanticFieldEngine:
    """
    F(x,y,t) = Σ_i conf_i × exp(−||p_i−(x,y)||² / 2σ_i²)
    Gaussian superposition. Divergence, curl, gradient, entanglement.
    """
    def analyze(self, frame_tags, metas, gs=GRID_SIZE):
        N = len(frame_tags)
        if N < 2:
            return self._empty()
        gx = np.linspace(0, 1, gs); gy = np.linspace(0, 1, gs)
        GX, GY = np.meshgrid(gx, gy)
        fields = np.zeros((N, gs, gs))
        for fi, ftags in enumerate(frame_tags):
            F = np.zeros((gs, gs))
            for t in ftags:
                cx = t.bbox[0] + t.bbox[2] / 2
                cy = t.bbox[1] + t.bbox[3] / 2
                sig = max(t.bbox[2], t.bbox[3]) * 0.5 + 0.05
                F += t.confidence * np.exp(-((GX - cx) ** 2 + (GY - cy) ** 2) / (2 * sig ** 2))
            fields[fi] = F
        mf = fields.mean(axis=0); vf = fields.var(axis=0)
        gy2, gx2 = np.gradient(mf)
        # Velocity field
        vfx = np.zeros((N, gs, gs)); vfy = np.zeros((N, gs, gs))
        for fi, ftags in enumerate(frame_tags):
            for t in ftags:
                cx = t.bbox[0] + t.bbox[2] / 2
                cy = t.bbox[1] + t.bbox[3] / 2
                sig = max(t.bbox[2], t.bbox[3]) * 0.5 + 0.05
                w = t.confidence * np.exp(-((GX - cx) ** 2 + (GY - cy) ** 2) / (2 * sig ** 2))
                vfx[fi] += w * t.velocity[0]
                vfy[fi] += w * t.velocity[1]
        mvx = vfx.mean(axis=0); mvy = vfy.mean(axis=0)
        dvxdx = np.gradient(mvx, axis=1); dvydy = np.gradient(mvy, axis=0)
        div = dvxdx + dvydy
        dvydx = np.gradient(mvy, axis=1); dvxdy = np.gradient(mvx, axis=0)
        curl = dvydx - dvxdy
        fp = mf.flatten(); fp /= (fp.sum() + 1e-12)
        fe = float(stats.entropy(fp + 1e-12))
        py2, px2 = np.unravel_index(mf.argmax(), mf.shape)
        pl = [round(float(gx[px2]), 3), round(float(gy[py2]), 3)]
        # Semantic entanglement
        by_id = {}
        for fi, ftags in enumerate(frame_tags):
            for t in ftags:
                if t.object_id not in by_id: by_id[t.object_id] = np.zeros(N)
                if fi < N: by_id[t.object_id][fi] = t.confidence
        oids = list(by_id.keys())[:6]; ent = {}
        for i in range(len(oids)):
            for j in range(i + 1, len(oids)):
                sa, sb = by_id[oids[i]], by_id[oids[j]]
                if sa.std() > 1e-6 and sb.std() > 1e-6:
                    mi = float(np.corrcoef(sa, sb)[0, 1]) ** 2
                    ent[f"{oids[i]}<>{oids[j]}"] = round(mi, 4)
        div_mean = float(np.mean(div)); div_max = float(np.max(np.abs(div)))
        curl_mean = float(np.mean(np.abs(curl))); curl_max = float(np.max(np.abs(curl)))
        return {
            "description": "F(x,y,t)=sum conf_i*exp(-||p_i-(x,y)||^2/2s_i^2)",
            "mean_field": [[round(float(v), 4) for v in row] for row in mf],
            "gradient_x_mean": [[round(float(v), 4) for v in row] for row in gx2],
            "gradient_y_mean": [[round(float(v), 4) for v in row] for row in gy2],
            "divergence": [[round(float(v), 4) for v in row] for row in div],
            "curl": [[round(float(v), 4) for v in row] for row in curl],
            "field_entropy": round(fe, 4),
            "peak_semantic_location": pl,
            "semantic_entanglement": dict(sorted(ent.items(), key=lambda x: -x[1])[:6]),
            "summary": {
                "divergence_mean": round(div_mean, 5), "divergence_max": round(div_max, 5),
                "curl_mean_abs": round(curl_mean, 5), "curl_max": round(curl_max, 5),
                "divergence_interpretation": (

            },
            "physics_interpretation": {
                "field": "Each object=particle=local field excitation. Fields superpose.",
                "gradient": "∇F: direction of max semantic change",
                "divergence": "∇·V>0: dispersing. <0: converging",

            }
        }

    def _empty(self):
        return {"description": "Insufficient data", "mean_field": [], "field_entropy": 0,
                "peak_semantic_location": [0.5, 0.5], "semantic_entanglement": {},
                "summary": {"divergence_mean": 0, "divergence_max": 0,
                            "curl_mean_abs": 0, "curl_max": 0,

class SemanticGeometryEngine:

    def analyze(self, frame_tags, metas, density):
        D = np.array(density); N = len(D)
        if N < 2:
            return self._empty()
        by_id = {}

            for t in ftags: by_id.setdefault(t.object_id, []).append(t)
        mean_vecs = {}
        for oid, tags in by_id.items():
            vecs = [t.semantic_state_vec for t in tags if t.semantic_state_vec]
            if vecs: mean_vecs[oid] = np.mean(vecs, axis=0)
        angles = {}; oids = list(mean_vecs.keys())[:10]

class TrigonometricEngine:
    """
    Phase analysis, angular velocity, harmonic decomposition.
    Trigonometric analysis of semantic motion and field rotation.
    """
    def analyze(self, density, frame_tags, metas):
        D = np.array(density, dtype=np.float64); N = len(D); fps = float(FPS_TARGET)
        if N < 4:
            return {"note": "Insufficient data for trigonometric analysis"}
        # Phase of each object's trajectory in semantic space
        by_id = {}
        for fi, ftags in enumerate(frame_tags):
            for t in ftags:
                by_id.setdefault(t.object_id, []).append(
                    (fi, t.bbox[0] + t.bbox[2] / 2, t.bbox[1] + t.bbox[3] / 2))
        # Angular analysis per object
        obj_angular = {}
        for oid, positions in by_id.items():
            if len(positions) < 3: continue
            # Compute angular velocity (change in direction of motion)
            angles_rad = []
            angular_vels = []
            for i in range(1, len(positions)):
                dx = positions[i][1] - positions[i - 1][1]
                dy = positions[i][2] - positions[i - 1][2]
                theta = math.atan2(dy, dx)
                angles_rad.append(theta)
            for i in range(1, len(angles_rad)):
                dtheta = angles_rad[i] - angles_rad[i - 1]
                # Normalize to [-pi, pi]
                dtheta = (dtheta + math.pi) % (2 * math.pi) - math.pi
                angular_vels.append(dtheta)
            if angular_vels:
                mean_omega = float(np.mean(angular_vels))
                max_omega = float(np.max(np.abs(angular_vels)))
                # Harmonic decomposition of angular velocity
                if len(angular_vels) > 4:
                    av_fft = np.fft.rfft(angular_vels)
                    av_power = np.abs(av_fft) ** 2
                    dom_harm = int(np.argmax(av_power[1:])) + 1
                else:
                    dom_harm = 0
                obj_angular[oid] = {
                    "mean_angular_velocity": round(mean_omega, 5),
                    "max_angular_velocity": round(max_omega, 5),
                    "dominant_harmonic": dom_harm,
                    "n_positions": len(positions),
                    "trajectory_curvature": round(
                        float(np.std(angular_vels)) if angular_vels else 0, 5)
                }
        # Phase coherence between objects
        phase_coherence = {}
        oids = list(obj_angular.keys())[:8]
        for i in range(len(oids)):
            for j in range(i + 1, len(oids)):
                oid_a, oid_b = oids[i], oids[j]
                pos_a = dict((p[0], (p[1], p[2])) for p in by_id[oid_a])
                pos_b = dict((p[0], (p[1], p[2])) for p in by_id[oid_b])
                common_frames = sorted(set(pos_a.keys()) & set(pos_b.keys()))
                if len(common_frames) < 3: continue
                # Relative angle between objects over time
                rel_angles = []
                for fi in common_frames:
                    dx = pos_b[fi][0] - pos_a[fi][0]
                    dy = pos_b[fi][1] - pos_a[fi][1]
                    rel_angles.append(math.atan2(dy, dx))
                # Phase coherence = 1 - circular variance
                sin_sum = sum(math.sin(a) for a in rel_angles) / len(rel_angles)
                cos_sum = sum(math.cos(a) for a in rel_angles) / len(rel_angles)
                R = math.sqrt(sin_sum ** 2 + cos_sum ** 2)
                phase_coherence[f"{oid_a}<>{oid_b}"] = round(R, 4)
        # Global D(t) phase analysis via Hilbert-like approach
        D_centered = D - D.mean()
        analytic = scipy_signal.hilbert(D_centered)
        inst_phase = np.angle(analytic)
        inst_freq = np.diff(np.unwrap(inst_phase)) * fps / (2 * math.pi)
        return {
            "object_angular_dynamics": dict(sorted(
                obj_angular.items(), key=lambda x: -abs(x[1]["mean_angular_velocity"]))[:8]),
            "phase_coherence": dict(sorted(
                phase_coherence.items(), key=lambda x: -x[1])[:6]),
            "global_phase": {
                "instantaneous_frequency_mean": round(float(np.mean(inst_freq)), 5) if len(inst_freq) > 0 else 0,
                "instantaneous_frequency_std": round(float(np.std(inst_freq)), 5) if len(inst_freq) > 0 else 0,
                "phase_stability": round(float(1.0 - np.std(inst_freq) / (np.mean(np.abs(inst_freq)) + 1e-12)), 4) if len(inst_freq) > 0 else 0,
            },
            "harmonic_summary": {
                "n_rotating_objects": sum(1 for v in obj_angular.values()
                                         if abs(v["mean_angular_velocity"]) > 0.01),
                "n_coherent_pairs": sum(1 for v in phase_coherence.values() if v > 0.7),
                "dominant_rotation": ("clockwise" if sum(
                    v["mean_angular_velocity"] for v in obj_angular.values()) < 0
                                      else "counterclockwise" if sum(
                    v["mean_angular_velocity"] for v in obj_angular.values()) > 0
                                      else "none")
            }
        }


class StatisticalEngine:
    """
    Comprehensive statistical inference engine.
    Distribution fitting, hypothesis tests, confidence intervals,
    regression analysis, correlation matrices.
    """
    def analyze(self, density, frame_tags, metas):
        D = np.array(density, dtype=np.float64); N = len(D)
        if N < 4:
            return {"note": "Insufficient data for statistical analysis"}
        # Distribution fitting
        # Test against multiple distributions
        norm_params = stats.norm.fit(D)
        norm_ks = stats.kstest(D, 'norm', args=norm_params)
        lognorm_D = D[D > 0]
        if len(lognorm_D) > 10:
            lognorm_params = stats.lognorm.fit(lognorm_D)
            lognorm_ks = stats.kstest(lognorm_D, 'lognorm', args=lognorm_params)
        else:
            lognorm_ks = type('obj', (object,), {'statistic': 1.0, 'pvalue': 0.0})()
        # Confidence intervals
        ci_95 = stats.t.interval(0.95, N - 1, loc=D.mean(), scale=stats.sem(D))
        # Stationarity test (simplified ADF-like)
        # Compute autocorrelation
        D_centered = D - D.mean()
        acf = np.correlate(D_centered, D_centered, mode='full')[N - 1:]
        acf /= (acf[0] + 1e-12)
        acf_vals = [round(float(acf[i]), 4) for i in range(min(20, len(acf)))]
        # Autocorrelation decay rate
        decay_lags = [i for i in range(1, min(50, len(acf))) if abs(acf[i]) < 0.1]
        decorrelation_time = decay_lags[0] if decay_lags else N
        # Object-level correlation matrix
        by_id = {}
        for fi, ftags in enumerate(frame_tags):
            for t in ftags:
                if t.object_id not in by_id: by_id[t.object_id] = np.zeros(N)
                if fi < N: by_id[t.object_id][fi] = t.confidence
        oids = list(by_id.keys())[:8]
        corr_matrix = {}
        for i in range(len(oids)):
            for j in range(i + 1, len(oids)):
                sa, sb = by_id[oids[i]], by_id[oids[j]]
                if sa.std() > 1e-6 and sb.std() > 1e-6:
                    r, p = stats.pearsonr(sa, sb)
                    corr_matrix[f"{oids[i]}<>{oids[j]}"] = {
                        "r": round(float(r), 4), "p": round(float(p), 6),
                        "significant": p < 0.05}
        # Trend analysis (Mann-Kendall simplified)
        concordant = sum(1 for i in range(N) for j in range(i + 1, N) if D[j] > D[i])
        discordant = sum(1 for i in range(N) for j in range(i + 1, N) if D[j] < D[i])
        n_pairs = N * (N - 1) / 2
        tau = (concordant - discordant) / n_pairs if n_pairs > 0 else 0
        # Changepoint detection (CUSUM-based)
        cusum = np.cumsum(D - D.mean())
        changepoints = []
        cusum_std = cusum.std() + 1e-12
        for i in range(10, N - 10):
            if abs(cusum[i]) > 2 * cusum_std:
                if not changepoints or i - changepoints[-1] > 20:
                    changepoints.append(i)
        return {
            "distribution": {
                "normal_fit": {"mean": round(float(norm_params[0]), 6),
                               "std": round(float(norm_params[1]), 6),
                               "ks_statistic": round(float(norm_ks.statistic), 5),
                               "ks_pvalue": round(float(norm_ks.pvalue), 5),
                               "is_normal": norm_ks.pvalue > 0.05},
                "best_fit": "normal" if norm_ks.pvalue > lognorm_ks.pvalue else "lognormal"
            },
            "confidence_intervals": {
                "mean_95": [round(float(ci_95[0]), 6), round(float(ci_95[1]), 6)],
                "std_error": round(float(stats.sem(D)), 6)
            },
            "autocorrelation": {
                "acf_values": acf_vals,
                "decorrelation_time_frames": decorrelation_time,
                "decorrelation_time_s": round(decorrelation_time / FPS_TARGET, 3)
            },
            "correlation_matrix": dict(sorted(
                corr_matrix.items(), key=lambda x: -abs(x[1]["r"]))[:10]),
            "trend": {
                "kendall_tau": round(float(tau), 5),
                "trend_direction": ("increasing" if tau > 0.1
                                    else "decreasing" if tau < -0.1
                                    else "no significant trend"),
                "trend_strength": round(abs(float(tau)), 4)
            },
            "changepoints": {
                "frames": changepoints[:10],
                "timestamps_s": [round(f / FPS_TARGET, 2) for f in changepoints[:10]],
                "n_changepoints": len(changepoints)
            }
        }

# SEMANTIC SIGNAL PROCESSING (SSP) — THE MONETIZATION LAYER
# ═══════════════════════════════════════════════════════════════════════════════
class SemanticSignalProcessor:
    """
    The core value engine. Identifies high-density, low-entropy semantic
    anchors in the topographic field. Calculates manipulation vectors for
    maximum value change with minimum semantic shift.

    The product is intelligence: semantic signals derived from the proxied
    topographical space. The trick is finding data points where:
      1. Semantic density is HIGH (many meanings concentrated)
      2. Entropy is LOW (the meaning is certain/immutable)
      3. A SMALL change in the tagged semantic data produces a LARGE
         change in the value of the reconstructed reality

    This is where money is made.
    """
    def analyze(self, frame_tags, metas, field_result, calc_result, density):
        D = np.array(density); N = len(D)
        if N < 2:
            return self._empty()

        # Step 1: Compute local semantic density and entropy per tag
        by_id = {}
        for fi, ftags in enumerate(frame_tags):
            for t in ftags:
                by_id.setdefault(t.object_id, []).append(t)

        # Step 2: Identify anchor candidates
        anchors = []
        for oid, tags in by_id.items():
            if len(tags) < 3: continue
            # Density: confidence × duration × centrality
            density_score = (sum(t.confidence for t in tags) / len(tags)
                             * math.log1p(len(tags))
                             * (1 + tags[0].centrality_score))
            # Entropy: variance in state vectors (lower = more immutable)
            vecs = [t.semantic_state_vec for t in tags if t.semantic_state_vec]
            if not vecs: continue
            vec_arr = np.array(vecs)
            entropy_score = float(np.mean(np.var(vec_arr, axis=0)))
            # Normalize
            density_norm = min(1.0, density_score / 5.0)
            entropy_norm = min(1.0, entropy_score * 10)
            # Anchor score: high density, low entropy
            anchor_score = density_norm * (1 - entropy_norm)
            # Store local density/entropy on tags
            for t in tags:
                t.semantic_density_local = round(density_norm, 4)
                t.semantic_entropy_local = round(entropy_norm, 4)
                t.ssp_anchor_score = round(anchor_score, 4)
            if (density_norm >= SSP_ANCHOR_DENSITY_THRESHOLD and
                    entropy_norm <= SSP_ANCHOR_ENTROPY_CEILING):
                mean_vec = np.mean(vec_arr, axis=0).tolist()
                cx = np.mean([t.bbox[0] + t.bbox[2] / 2 for t in tags])
                cy = np.mean([t.bbox[1] + t.bbox[3] / 2 for t in tags])
                anchors.append(SemanticAnchor(
                    anchor_id=f"ssp_{oid}", object_id=oid,
                    frame_index=tags[0].frame_index,
                    density=density_norm, entropy=entropy_norm,
                    field_position=[round(float(cx), 4), round(float(cy), 4)],
                    state_vector=mean_vec,
                    manipulation_potential=0.0,
                    prime_alignment=0.0
                ))

        # Step 3: Calculate manipulation potential for each anchor
        # How much does the field change if we shift this anchor slightly?
        field_data = field_result.get("mean_field", [])
        if field_data and anchors:
            field_arr = np.array(field_data)
            for anchor in anchors:
                # Perturbation analysis
                fx = int(anchor.field_position[0] * (GRID_SIZE - 1))
                fy = int(anchor.field_position[1] * (GRID_SIZE - 1))
                fx = max(1, min(GRID_SIZE - 2, fx))
                fy = max(1, min(GRID_SIZE - 2, fy))
                # Local gradient magnitude at anchor position
                if fy < field_arr.shape[0] and fx < field_arr.shape[1]:
                    local_grad = math.sqrt(
                        (field_arr[fy, min(fx + 1, GRID_SIZE - 1)] -
                         field_arr[fy, max(fx - 1, 0)]) ** 2 +
                        (field_arr[min(fy + 1, GRID_SIZE - 1), fx] -
                         field_arr[max(fy - 1, 0), fx]) ** 2)
                    # Manipulation potential = gradient * density / entropy
                    anchor.manipulation_potential = round(
                        local_grad * anchor.density / (anchor.entropy + 0.01), 4)

        # Step 4: Value estimation (Calculus of Value integration)
        # V(t) = ∫U(s)·w(s)ds where U = utility, w = recognition
        total_utility = calc_result.get("integrals", {}).get("total_semantic_mass", 0)
        recognition_weight = len(anchors) / max(len(by_id), 1)  # fraction that are anchors
        value_integral = total_utility * recognition_weight

        # Attention flow A(t) = d/dt Engagement
        # Approximated by D'(t) — rate of change of semantic density
        d1 = calc_result.get("derivatives", {}).get("d1_mean_abs", 0)
        attention_flow = d1

        # Revenue proxy R = ∫A(t)·V(t)dt
        revenue_proxy = attention_flow * value_integral

        # Step 5: Cross-medium translation potential
        # Can this semantic structure be expressed in a different medium?
        constraint_tightness = 0.5
        if calc_result:
            cv = calc_result.get("density", {}).get("coefficient_of_variation", 0.5)
            constraint_tightness = 1.0 - min(1.0, cv)

        # Sort anchors by manipulation potential
        anchors.sort(key=lambda a: -a.manipulation_potential)

        return {
            "anchors": [
                {
                    "anchor_id": a.anchor_id, "object_id": a.object_id,
                    "density": a.density, "entropy": a.entropy,
                    "field_position": a.field_position,
                    "manipulation_potential": a.manipulation_potential,
                    "prime_alignment": a.prime_alignment,
                    "interpretation": (
                        "GOLD ANCHOR: highest density, lowest entropy, maximum manipulation leverage"
                        if i == 0 and a.manipulation_potential > 0.5
                        else "HIGH VALUE: strong anchor with significant manipulation potential"
                        if a.manipulation_potential > 0.3
                        else "STABLE ANCHOR: reliable but moderate manipulation potential")
                } for i, a in enumerate(anchors[:10])
            ],
            "n_anchors": len(anchors),
            "n_total_objects": len(by_id),
            "anchor_density_ratio": round(len(anchors) / max(len(by_id), 1), 4),
            "value_metrics": {
                "total_utility": round(total_utility, 4),
                "recognition_weight": round(recognition_weight, 4),
                "value_integral": round(value_integral, 4),
                "attention_flow": round(attention_flow, 6),
                "revenue_proxy": round(revenue_proxy, 6),
                "calculus_of_value": {
                    "V_t": round(value_integral, 4),
                    "A_t": round(attention_flow, 6),
                    "R": round(revenue_proxy, 6),
                    "interpretation": (
                        "Value = Utility × Recognition. Revenue = Attention × Value. "
                        f"This semantic structure has {'HIGH' if revenue_proxy > 0.01 else 'MODERATE' if revenue_proxy > 0.001 else 'LOW'} "
                        "monetization potential through semantic signal extraction.")
                }
            },
            "cross_medium_translation": {
                "constraint_tightness": round(constraint_tightness, 4),
                "translatable": constraint_tightness > 0.4,
                "recommended_output_media": self._recommend_media(anchors, constraint_tightness),
                "note": ("The semantic structure can be reconstructed in a different medium "
                         "than the original. High constraint tightness means fewer possible "
                         "realities, making translation more precise.")
            },
            "semantic_manipulation_guide": {
                "optimal_anchor": anchors[0].anchor_id if anchors else None,
                "optimal_shift_direction": "along gradient" if anchors else None,
                "expected_value_change": round(
                    anchors[0].manipulation_potential * 10 if anchors else 0, 4),
                "precision_required": SSP_MANIPULATION_PRECISION,
                "methodology": (
                    "1. Identify high-density low-entropy anchor in semantic field\n"
                    "2. Calculate gradient at anchor position\n"
                    "3. Apply minimal semantic shift along gradient direction\n"

# PRIME RESONANCE ENGINE
# ═══════════════════════════════════════════════════════════════════════════════
class PrimeResonanceEngine:
    """
    Integrates the Prime Translation Framework for immutable scaffolding.
    
    Primes are irreducible — they carry exactly one unit of structural truth.
    When rebuilding reality from semantic space, we need immutable anchors
    that cannot be further decomposed. Primes provide this scaffolding.
    
    The process:
    1. Map semantic density D(t) to a gap sequence (analogous to prime gaps)
    2. Compute normalized gap anomalies
    3. Multi-scale smoothing to reveal resonance structure
    4. Build invariant T-vectors (10D)
    5. Classify semantic states as Ω (constructive), Λ (destructive), N (neutral)
    6. Identify prime-like scaffolding points in the semantic field
    7. These scaffolding points anchor reality reconstruction
    """
    def analyze(self, density, frame_tags, ssp_result):
        D = np.array(density, dtype=np.float64); N = len(D)
        if N < 10:
            return self._empty()

        # Step 1: Derive gap sequence from semantic density
        # Gaps = differences between consecutive density values
        gaps = np.diff(D)
        if len(gaps) < 5:
            return self._empty()

        mu_g = float(np.mean(gaps))
        sigma_g = float(np.std(gaps) + 1e-12)
        ng = (gaps - mu_g) / sigma_g  # normalized gap anomaly

        # Step 2: Multi-scale smoothing (5 incommensurate widths)
        A_w = {}
        for w in PRIME_SMOOTHING_WIDTHS:
            sig = w / 2.355
            kernel = np.exp(-np.arange(-3 * sig, 3 * sig + 1) ** 2 / (2 * sig ** 2))
            kernel /= kernel.sum()
            smoothed = np.convolve(ng, kernel, mode='same')
            A_w[w] = smoothed

        # Step 3: Compute Order Invariant (OI) — multi-scale synchrony
        OI = np.zeros(len(ng))
        for w in PRIME_SMOOTHING_WIDTHS:
            OI += A_w[w] ** 2
        OI = np.sqrt(OI)

        # Step 4: Identify prime-like scaffolding points
        # Points where the semantic structure is most irreducible
        # (highest OI, lowest local entropy)
        scaffold_points = []
        window = 10
        for i in range(window, len(ng) - window):
            local_entropy = float(np.std(ng[i - window:i + window]))
            oi_val = float(OI[i])
            # Prime-like criterion: high synchrony, low entropy
            if oi_val > np.percentile(OI, 75) and local_entropy < np.percentile(
                    [np.std(ng[max(0, j - window):j + window])
                     for j in range(window, len(ng) - window, 5)], 25):
                scaffold_points.append({
                    "frame": i,
                    "timestamp_s": round(i / FPS_TARGET, 3),
                    "OI": round(oi_val, 5),
                    "local_entropy": round(local_entropy, 5),
                    "density_value": round(float(D[i]), 5),
                    "immutability_score": round(oi_val / (local_entropy + 0.01), 4)
                })

        # Step 5: Classify semantic states
        omega_frames = []  # constructive synchronization
        lambda_frames = []  # destructive synchronization
        for i in range(len(OI)):
            if OI[i] > np.percentile(OI, 80):
                if ng[i] > 0:
                    omega_frames.append(i)
                else:
                    lambda_frames.append(i)

        # Step 6: Compute resonance balance
        # How many immutable scaffolding points do we need to rebuild reality?
        total_info = float(np.sum(np.abs(ng)))  # total information content
        scaffold_info = sum(abs(ng[sp["frame"]]) for sp in scaffold_points)
        coverage = scaffold_info / (total_info + 1e-12)

        # Step 7: Align SSP anchors with prime scaffolding
        ssp_anchors = ssp_result.get("anchors", [])
        alignments = []
        for anchor in ssp_anchors[:5]:
            # Find nearest scaffold point
            anchor_frame = anchor.get("frame_index", 0) if isinstance(anchor, dict) else 0
            nearest_scaffold = None
            min_dist = float('inf')
            for sp in scaffold_points:
                d = abs(sp["frame"] - anchor_frame)
                if d < min_dist:
                    min_dist = d
                    nearest_scaffold = sp
            if nearest_scaffold:
                alignments.append({
                    "anchor_id": anchor.get("anchor_id", ""),
                    "nearest_scaffold_frame": nearest_scaffold["frame"],
                    "distance_frames": min_dist,
                    "alignment_score": round(1.0 / (1 + min_dist / 10), 4)
                })

        # Sort scaffolding by immutability
        scaffold_points.sort(key=lambda x: -x["immutability_score"])

        return {
            "gap_statistics": {
                "mean_gap": round(mu_g, 6),
                "std_gap": round(sigma_g, 6),
                "n_gaps": len(gaps)
            },
            "order_invariant": {

# .THRU VIDEO RENDERER
# ═══════════════════════════════════════════════════════════════════════════════
def render_thru_video(frame_tags, metas, density, wave_result, field_result, out_path: str):
    """Pure semantic representation. No original footage."""
    try:
        import cv2
    except ImportError:
        return

    N = len(frame_tags)
    fps = FPS_TARGET
    domf = wave_result.get("fourier", {}).get("dominant_frequency_hz", 1.0)
    dom_period_frames = max(1, round(fps / max(domf, 0.01)))
    fourcc = cv2.VideoWriter_fourcc(*'mp4v')
    writer = cv2.VideoWriter(out_path, fourcc, fps, (THRU_W, THRU_H))
    D_arr = np.array(density)
    field_data = np.array(field_result.get("mean_field", [[0] * GRID_SIZE] * GRID_SIZE))
    field_norm = (field_data - field_data.min()) / (field_data.max() - field_data.min() + 1e-12)
    div_data = np.array(field_result.get("divergence", [[0] * GRID_SIZE] * GRID_SIZE))
    curl_data = np.array(field_result.get("curl", [[0] * GRID_SIZE] * GRID_SIZE))
    PH = THRU_H - 80

    for fi in range(N):
        canvas = np.zeros((THRU_H, THRU_W, 3), dtype=np.uint8)
        density_val = float(D_arr[fi]) if fi < len(D_arr) else 0.5
        pulse_frac = (fi % dom_period_frames) / dom_period_frames
        pulse_ac = 0.5 + 0.5 * math.sin(pulse_frac * 2 * math.pi)
        pulse_op = 0.5 + 0.5 * math.sin((1 - pulse_frac) * 2 * math.pi + math.pi)

        # Background: semantic field heatmap
        for gy in range(GRID_SIZE):
            for gx_i in range(GRID_SIZE):
                fv = float(field_norm[gy][gx_i]) if gy < len(field_norm) and gx_i < len(field_norm[gy]) else 0
                dv = float(div_data[gy][gx_i]) if gy < len(div_data) and gx_i < len(div_data[gy]) else 0
                cv3 = float(curl_data[gy][gx_i]) if gy < len(curl_data) and gx_i < len(curl_data[gy]) else 0
                x1b = int(gx_i * THRU_W / GRID_SIZE); x2b = int((gx_i + 1) * THRU_W / GRID_SIZE)
                y1b = int(gy * PH / GRID_SIZE); y2b = int((gy + 1) * PH / GRID_SIZE)
                b = int(fv * 25 + max(0, -dv) * 40)
                g = int(fv * 20 + abs(cv3) * 30)
                r = int(fv * 15 + max(0, dv) * 35)
                canvas[y1b:y2b, x1b:x2b] = (min(255, b), min(255, g), min(255, r))

        ftags = frame_tags[fi] if fi < N else []
        # Draw relations
        for tag in ftags:
            cx1 = int((tag.bbox[0] + tag.bbox[2] / 2) * THRU_W)
            cy1 = int((tag.bbox[1] + tag.bbox[3] / 2) * PH)
            for rel in tag.relations[:2]:
                rt = next((t for t in ftags if t.object_id == rel.object_id), None)
                if not rt: continue
                cx2 = int((rt.bbox[0] + rt.bbox[2] / 2) * THRU_W)
                cy2 = int((rt.bbox[1] + rt.bbox[3] / 2) * PH)
                cv2.line(canvas, (cx1, cy1), (cx2, cy2), (40, 50, 80), 1, cv2.LINE_AA)

        # Draw objects as dots
        for tag in ftags:
            cx = int((tag.bbox[0] + tag.bbox[2] / 2) * THRU_W)
            cy = int((tag.bbox[1] + tag.bbox[3] / 2) * PH)
            base_r = max(8, int(tag.confidence * 20 + tag.centrality_score * 1.5))
            color = cat_bgr(tag.class_label)
            if tag.phonon_mode == "acoustic":
                r = int(base_r * (0.85 + pulse_ac * 0.3))
            elif tag.phonon_mode == "optical":
                r = int(base_r * (0.85 + pulse_op * 0.3))
            else:
                r = base_r
            # SSP anchor glow
            if tag.ssp_anchor_score > 0.5:
                cv2.circle(canvas, (cx, cy), r + 18, (0, 255, 200), 2, cv2.LINE_AA)
            # Gold anchor
            if tag.is_anchor:
                cv2.circle(canvas, (cx, cy), r + 14, (0, 215, 255), 3, cv2.LINE_AA)
            elif tag.is_high_confidence:
                cv2.circle(canvas, (cx, cy), r + 9, (200, 50, 200), 2, cv2.LINE_AA)
            cv2.circle(canvas, (cx, cy), r, color, -1, cv2.LINE_AA)
            # Velocity arrow
            vx, vy = tag.velocity
            if abs(vx) + abs(vy) > 0.005:
                ex = cx + int(vx * THRU_W * 3); ey = cy + int(vy * PH * 3)
                ex = max(0, min(THRU_W - 1, ex)); ey = max(0, min(PH - 1, ey))
                cv2.arrowedLine(canvas, (cx, cy), (ex, ey),
                                tuple(min(255, c + 80) for c in color), 1, cv2.LINE_AA)

        # HUD
        cv2.rectangle(canvas, (0, PH), (THRU_W, THRU_H), (8, 10, 16), -1)
        if len(D_arr) > 1:
            for i in range(1, THRU_W):
                di = min(int(i * len(D_arr) / THRU_W), len(D_arr) - 1)
                dip = max(0, int((i - 1) * len(D_arr) / THRU_W))
                v1 = float(D_arr[dip]); v2 = float(D_arr[di])
                y1h = THRU_H - 4 - int(v1 * 72); y2h = THRU_H - 4 - int(v2 * 72)
                cv2.line(canvas, (i - 1, y1h), (i, y2h), (60, 170, 200), 1, cv2.LINE_AA)
            fx = int(fi * THRU_W / max(len(D_arr) - 1, 1))
            cv2.line(canvas, (fx, PH + 2), (fx, THRU_H - 2), (20, 160, 232), 2)
        ts = f"T+{int(fi / fps / 60):02d}:{int(fi / fps) % 60:02d}"
        cv2.putText(canvas, f"THRUPUT v4 {ts} D={density_val:.3f}",
                    (12, PH + 20), cv2.FONT_HERSHEY_PLAIN, 1.0, (60, 90, 120), 1, cv2.LINE_AA)
        cv2.putText(canvas, f"n={len(ftags)} objs",
                    (12, PH + 38), cv2.FONT_HERSHEY_PLAIN, 0.9, (40, 60, 80), 1, cv2.LINE_AA)
        writer.write(canvas)
    writer.release()

# DIMENSIONAL NORMALIZER
# ═══════════════════════════════════════════════════════════════════════════════
def _dimensional_normalize(frame_tags: list, medium: str) -> list:
    """
    Normalize all mediums to canonical semantic coordinates.
    Solves: audio spectral → same field as YOLO bbox → same field as text tokens.
    Without this, DynamicalEngine/CalculusEngine compute on mixed distributions = noise.
    
    Canonical semantic coordinates:
      position = semantic meaning position (not spatial/spectral/textual)
      confidence = reliability-adjusted per medium
      velocity = semantic rate of change (not pixel/spectral/token displacement)
    """
    # Medium-specific confidence scaling (audio/language less reliable than video/image)
    conf_scale = {"video": 1.0, "image": 1.0, "audio": 0.8, "language": 0.75}
    noise_floor = {"video": 0.05, "image": 0.05, "audio": 0.1, "language": 0.12}
    vel_scale = {"video": 1.0, "image": 0.0, "audio": 0.3, "language": 0.2}
    
    cs = conf_scale.get(medium, 1.0)
    nf = noise_floor.get(medium, 0.05)
    vs = vel_scale.get(medium, 1.0)
    
    # Collect all positions for z-score normalization (non-video only)
    if medium != "video":
        all_cx, all_cy = [], []
        for ftags in frame_tags:
            for t in ftags:
                all_cx.append(t.bbox[0] + t.bbox[2] / 2)
                all_cy.append(t.bbox[1] + t.bbox[3] / 2)
        if all_cx:
            cx_m, cx_s = np.mean(all_cx), np.std(all_cx) + 1e-8
            cy_m, cy_s = np.mean(all_cy), np.std(all_cy) + 1e-8
        else:
            cx_m, cx_s, cy_m, cy_s = 0.5, 1.0, 0.5, 1.0
    
    for ftags in frame_tags:
        for t in ftags:
            # Scale confidence by medium reliability
            t.confidence = max(nf, t.confidence * cs)
            # Scale velocity by medium type
            t.velocity = [v * vs for v in t.velocity]
            t.acceleration = [a * vs for a in t.acceleration]
            # Normalize non-video positions to semantic coordinates
            if medium != "video":
                cx = t.bbox[0] + t.bbox[2] / 2
                cy = t.bbox[1] + t.bbox[3] / 2
                cx_n = max(0.05, min(0.95, 0.5 + (cx - cx_m) / (4 * cx_s)))
                cy_n = max(0.05, min(0.95, 0.5 + (cy - cy_m) / (4 * cy_s)))
                w, h = t.bbox[2], t.bbox[3]
                t.bbox = [cx_n - w / 2, cy_n - h / 2, w, h]
    return frame_tags

# INTELLIGENCE PRODUCT GENERATOR
# ═══════════════════════════════════════════════════════════════════════════════
def _generate_intelligence_products(bundle: dict, frame_tags: list,
                                     ssp: dict, prime: dict, medium: str) -> dict:
    """
    THE MONEY ENGINE.
    
    Takes semantic analysis → produces monetizable intelligence products.
    Input cost: ~$0.30 (API). Output value: $100-5000 per product.
    
    The semantic structure costs nothing to derive.
    The intelligence it produces costs thousands to obtain otherwise.
    Value ratio: 100x-10000x.
    """
    api_cost = 0.30
    products = []
    calc = bundle.get("calculus", {})
    simple = bundle.get("simple", {})
    lexg = bundle.get("lexguard", {})
    dynam = bundle.get("dynamical", {})
    geom = bundle.get("geometry", {})
    
    # Product 1: Semantic Audit
    deception = simple.get("deception_indicators", [])
    harmony = lexg.get("harmony", {}).get("mean", 0)
    policy = lexg.get("socp", {}).get("policy", "PROCEED")
    hurst = calc.get("topology", {}).get("hurst_exponent", 0.5)
    audit_value = 200
    if deception: audit_value *= 3
    if lexg.get("pars_elevated", {}).get("n_tier2_flags", 0) > 0: audit_value *= 5
    if ssp.get("n_anchors", 0) > 5: audit_value *= 2
    products.append({
        "type": "semantic_audit", "title": f"Semantic Audit: {medium.title()}",
        "summary": (f"{'⚠ DECEPTION DETECTED. ' if deception else ''}"
                    f"Harmony={harmony:.4f} ({policy}). H={hurst:.4f}. "
                    f"{ssp.get('n_anchors', 0)} anchors."),
        "estimated_value_usd": round(audit_value, 2),
        "cost_usd": api_cost, "value_ratio": round(audit_value / api_cost, 0),
        "confidence": min(0.95, 0.5 + abs(harmony) * 0.2 + (0.2 if deception else 0))
    })
    
    # Product 2: Behavioral Intelligence
    attractor = dynam.get("state_space", {}).get("attractor_type", "")
    bifs = dynam.get("bifurcations", {}).get("n_bifurcations", 0)
    beh_value = 500
    if "chaotic" in attractor.lower(): beh_value *= 2
    if bifs > 0: beh_value *= 1.5
    products.append({
        "type": "behavioral_intelligence",
        "title": "Behavioral Pattern Intelligence",
        "summary": f"Attractor: {attractor}. {bifs} regime shifts. "
                   f"{'Chaotic—short prediction horizon' if 'chaotic' in attractor.lower() else 'Stable—predictable'}.",
        "estimated_value_usd": round(beh_value, 2),
        "cost_usd": api_cost, "value_ratio": round(beh_value / api_cost, 0),
        "confidence": min(0.9, 0.4 + (0.2 if attractor else 0) + (0.1 if bifs > 0 else 0))
    })
    
    # Product 3: Scenario Simulation (only if scaffolding sufficient)
    coverage = prime.get("reality_reconstruction", {}).get("scaffold_coverage", 0)
    tightness = geom.get("constraint_space", {}).get("constraint_tightness", 0)
    anchors = ssp.get("anchors", [])
    if coverage > 0.2 and anchors:
        manip = anchors[0].get("manipulation_potential", 0) if anchors else 0
        scen_value = 1000
        if tightness > 0.5: scen_value *= 3
        if coverage > 0.3: scen_value *= 2
        if manip > 0.5: scen_value *= 5
        products.append({
            "type": "scenario_simulation",
            "title": "Synthetic Scenario Intelligence",
            "summary": f"Scaffold coverage={coverage:.1%}. Tightness={tightness:.4f}. "
                       f"Manipulation potential={manip:.4f}. "
                       f"{'Viable for reliable simulation.' if coverage > 0.3 else 'Limited fidelity.'}",
            "estimated_value_usd": round(scen_value, 2),
            "cost_usd": api_cost * 2, "value_ratio": round(scen_value / (api_cost * 2), 0),
            "confidence": min(0.85, coverage * 0.4 + tightness * 0.3 + (0.2 if manip > 0.3 else 0))
        })
    
    # Product 4: Vulnerability Scan
    harm_min = lexg.get("harmony", {}).get("min", 0)
    redirect_pct = lexg.get("socp", {}).get("redirect_pct", 0)
    lyap_stable = lexg.get("lyapunov", {}).get("safety_invariant", True)
    n_vulns = len(deception) + (1 if harm_min < -0.2 else 0) + (1 if not lyap_stable else 0)
    vuln_value = 300
    if deception: vuln_value *= 4
    if harm_min < -0.2: vuln_value *= 2
    products.append({
        "type": "vulnerability_scan",
        "title": "Semantic Vulnerability Analysis",
        "summary": f"{n_vulns} vulnerabilities. Harmony min={harm_min:.4f}. "
                   f"Safety {'STABLE' if lyap_stable else 'UNSTABLE'}.",
        "estimated_value_usd": round(vuln_value, 2),
        "cost_usd": api_cost, "value_ratio": round(vuln_value / api_cost, 0),
        "confidence": min(0.9, 0.5 + n_vulns * 0.1),
        "severity": "CRITICAL" if n_vulns >= 3 else "HIGH" if n_vulns >= 2 else "MODERATE"
    })
    
    products.sort(key=lambda p: -p["value_ratio"])
    total_value = sum(p["estimated_value_usd"] for p in products)
    total_cost = sum(p["cost_usd"] for p in products)
    
    return {
        "products": products,
        "portfolio": {
            "total_value_usd": round(total_value, 2),
            "total_cost_usd": round(total_cost, 2),
            "portfolio_ratio": round(total_value / max(total_cost, 0.01), 0),
            "n_products": len(products),
            "top_product": products[0]["type"] if products else None,
        },
        "business_model": {
            "type": "Intelligence-as-a-Service",
            "input_cost": "Free (public content) or client-provided",
            "processing_cost": f"~${total_cost:.2f}",
            "output_value": f"~${total_value:.2f}",
            "margin": f"{((total_value - total_cost) / max(total_value, 1) * 100):.0f}%",
            "delivery": "SaaS API, PDF reports, or consulting engagements"
        }
    }

# PIPELINE
# ═══════════════════════════════════════════════════════════════════════════════
async def run_pipeline(job_id: str, in_path: str, medium: str, public: bool):
    def upd(stage, pct, **kw):
        vals = {"stage": stage, "progress": pct, "status": "running"}
        vals.update(kw)
        sc = ", ".join(f"{k}=?" for k in vals)
        with get_db() as conn:
            conn.execute(f"UPDATE jobs SET {sc} WHERE id=?", [*vals.values(), job_id])
    try:
        # ─── INGEST ───────────────────────────────────────────
        upd("INGEST", 2)
        if medium == MEDIUM_VIDEO:
            frame_tags, frames = convert_video(in_path)
        elif medium == MEDIUM_AUDIO:
            frame_tags, frames = convert_audio(in_path)
        elif medium == MEDIUM_IMAGE:
            frame_tags, frames = convert_image(in_path)
        elif medium == MEDIUM_LANGUAGE:
            frame_tags, frames = convert_language(in_path)
        else:
            frame_tags, frames = convert_video(in_path)

        N = len(frame_tags)
        frame_metas = []
        for fi, ftags in enumerate(frame_tags):
            meta = build_frame_meta(
                frames[fi] if fi < len(frames) else None, ftags, fi, medium)
            frame_metas.append(meta)
        upd("INGEST", 6, frames=N)

        # ─── DETECT/TRACK ────────────────────────────────────
        upd("TRACK", 15)
        frame_tags = tracking_pass(frame_tags)
        frame_tags = relations_pass(frame_tags)
        frame_tags = temporal_pass(frame_tags)
        frame_tags = semantic_state_pass(frame_tags)
        if medium == MEDIUM_VIDEO and frames:
            frame_tags = ocr_pass(frame_tags, frames)
        tags_total = sum(len(t) for t in frame_tags)

        # ─── DIMENSIONAL NORMALIZATION ────────────────────────
        # Solves coercion between audio (spectral) / video (bbox) / language (tokens)
        # All mediums normalized to canonical semantic coordinates before engines
        frame_tags = _dimensional_normalize(frame_tags, medium)

        density = [m.semantic_density for m in frame_metas]
        upd("TRACK", 30, tags_total=tags_total,
            density_peak=float(max(density)) if density else 0)

        # ─── MATH ENGINES ────────────────────────────────────
        upd("MATH", 35)
        wave_r = WaveEngine().analyze(density, frame_tags, frame_metas)
        upd("MATH", 40)
        calc_r = CalculusEngine().analyze(density, frame_tags, frame_metas)
        graph_r = GraphEngine().analyze(frame_tags)
        causal_r = CausalEngine().analyze(frame_tags)
        dynam_r = DynamicalEngine().analyze(density)
        upd("MATH", 50)
        tensor_r = TensorEngine().analyze(frame_tags)
        audio_r = AudioEngine().analyze(frame_metas, None)
        simple_r = SimpleEngine().analyze(density, frame_tags, frame_metas, wave_r, calc_r)
        field_r = SemanticFieldEngine().analyze(frame_tags, frame_metas)
        angle_r = SemanticGeometryEngine().analyze(frame_tags, frame_metas, density)
        upd("MATH", 58)
        trig_r = TrigonometricEngine().analyze(density, frame_tags, frame_metas)
        stat_r = StatisticalEngine().analyze(density, frame_tags, frame_metas)

        # ─── LEXGUARD ────────────────────────────────────────
        upd("LEXGUARD", 65)
        frame_metas = lexguard_pass(frame_tags, frame_metas)
        lexg_r = LexGuardEngine().analyze(frame_tags, frame_metas, calc_r, wave_r)

        # ─── SEMANTIC SIGNAL PROCESSING ──────────────────────
        upd("SSP", 72)
        ssp_r = SemanticSignalProcessor().analyze(
            frame_tags, frame_metas, field_r, calc_r, density)

        # ─── PRIME RESONANCE ─────────────────────────────────
        upd("PRIME", 78)
        prime_r = PrimeResonanceEngine().analyze(density, frame_tags, ssp_r)

        # ─── INTELLIGENCE PRODUCTS ───────────────────────────
        upd("INTELLIGENCE", 80)
        math_bundle = dict(
            wave=wave_r, calculus=calc_r, graph=graph_r, causal=causal_r,
            dynamical=dynam_r, tensor=tensor_r, audio=audio_r, simple=simple_r,
            field=field_r, geometry=angle_r, lexguard=lexg_r,
            ssp=ssp_r, prime=prime_r, trigonometric=trig_r, statistical=stat_r)
        intel_r = _generate_intelligence_products(math_bundle, frame_tags, ssp_r, prime_r, medium)

        # ─── AI FINDINGS ─────────────────────────────────────
        upd("CONCLUDE", 84)
        math_bundle["intelligence"] = intel_r
        conclusions = await generate_findings_ai(
            math_bundle, frame_tags, frame_metas, job_id, medium)

        # ─── RENDER ──────────────────────────────────────────
        upd("RENDER", 88)
        thru_path = str(OUTPUT_DIR / f"{job_id}.thru.mp4")
        if medium in (MEDIUM_VIDEO, MEDIUM_IMAGE):
            render_thru_video(frame_tags, frame_metas, density, wave_r, field_r, thru_path)
        else:
            # For audio/language, still render a semantic visualization
            render_thru_video(frame_tags, frame_metas, density, wave_r, field_r, thru_path)

        # ─── PACKAGE ─────────────────────────────────────────
        upd("PACKAGE", 93)
        prov = build_provenance(job_id, in_path, frame_tags, medium)
        semantic = {
            "thruput_version": "4.0", "job_id": job_id, "medium": medium,
            "provenance": prov,
            "fps": FPS_TARGET, "frames": N,
            "duration_s": round(N / FPS_TARGET, 3),
            "tags_total": tags_total, "density": density,
            "tag_inventory": build_tag_inventory(frame_tags),
            "wave": wave_r, "calculus": calc_r, "graph": graph_r,
            "causal": causal_r, "dynamical": dynam_r, "tensor": tensor_r,
            "audio": audio_r, "simple": simple_r, "field": field_r,
            "geometry": angle_r, "trigonometric": trig_r,
            "statistical": stat_r, "lexguard": lexg_r,
            "ssp": ssp_r, "prime": prime_r,
            "intelligence_products": intel_r,
            "conclusions": conclusions,
        }
        json_path = str(OUTPUT_DIR / f"{job_id}.json")
        with open(json_path, "w") as f:
            json.dump(semantic, f, indent=2, default=str)

        with get_db() as conn:
            conn.execute(
                "UPDATE jobs SET status='done',completed_at=?,thru_path=?,"
                "json_path=?,progress=100,stage='DONE',tags_total=? WHERE id=?",
                (time.strftime("%Y-%m-%dT%H:%M:%SZ"), thru_path,
                 json_path, tags_total, job_id))
        if public:
            index_corpus(job_id, semantic, medium)

    except Exception as e:
        import traceback; traceback.print_exc()
        with get_db() as conn:
            conn.execute("UPDATE jobs SET status='error',error=? WHERE id=?",
                         (str(e)[:2000], job_id))
    finally:
        p = Path(in_path)
        if p.exists():
            p.unlink(missing_ok=True)

Ontology: # THRUPUT — Foundational Ontology

## Part I — The Two Spaces

### 1.1 Reality Space (R-Space)

R-Space is the physical world as recorded by a medium. It has these properties:

**Axes:** Three spatial dimensions (x, y, z) + time (t). For a video, we observe a 2D projection (x, y) indexed by t. For audio, we observe amplitude over t. For text, we observe a sequence of symbols over t (reading order). For an image, we observe (x, y) at a single t.

**Objects in R-Space** are physical. A horse is a horse. A person is a person. They have mass, position, velocity, color, temperature. They obey physical laws — gravity, conservation of momentum, thermodynamics. They cannot teleport. They cannot be in two places at once. They exist whether observed or not.

**Time in R-Space** is singular, irreversible, and continuous. Events happen once. The arrow of time points one direction. You cannot rewind reality. Cause precedes effect — always. The speed of time is fixed at 1 second per second.

**Measurement in R-Space** is lossy. A camera captures a projection. A microphone captures pressure waves. Text captures symbolic encoding of thought. Every medium loses information from reality. A video of a horse does not contain the horse's temperature, smell, or the grass it stands on outside the frame. This loss is permanent and unrecoverable from the medium alone.

**The key constraint of R-Space:** Everything in it is particular. This specific horse at this specific time in this specific field. No generalization exists in R-Space. Generalization is a property of minds (and semantic space).

---

### 1.2 Semantic Space (S-Space)

S-Space is the topological field we construct from tagged observations. It has these properties:

**Axes:** The same dimensional structure as R-Space for that medium — (x, y, t) for video, (frequency, t) for audio, (position_in_sequence, t) for text — BUT the values on these axes are not physical measurements. They are SEMANTIC COORDINATES. 

The x-position of a tag in S-Space does not mean "30% from the left edge of the frame." It means "this is where this unit of meaning lives relative to other units of meaning." The coordinate system is inherited from R-Space's structure but the VALUES are derived, not measured.

**Objects in S-Space** are tags. A tag is a symbolic label attached to an observation: "horse", "person_walking", "profanity", "arrest." Tags are NOT the objects they describe. The tag "horse" has zero mass, no color, no temperature. It is a pure semantic unit — a carrier of meaning and nothing else.

**CRITICAL LAW 1 — The Tag-Object Bond:**
A tag and its source object are permanently bonded at the moment of tagging. The tag "horse" assigned to a specific detected horse at frame 47 ALWAYS refers to that exact horse at that exact moment. This bond is:
- Immutable (cannot be broken)
- Directional (object → tag, never tag → object)
- One-to-many (one object can have many tags: "horse", "brown", "grazing", "large")
- Non-invertible in isolation (the tag "horse" does not uniquely identify which horse or which frame without the bond metadata)

**Time in S-Space** is fundamentally different from R-Space:

Time in S-Space is AN AXIS, not a flow. It is a dimension you can move along freely — forward, backward, or you can view the entire timeline simultaneously. This is because S-Space is a database, not a physical system. The entire tagged history exists as a static structure that you can query at any point.

This is the single most important difference between the two spaces. In R-Space, you experience time. In S-Space, you OBSERVE time as a completed dimension, like looking at a map instead of walking a road.

**CRITICAL LAW 2 — Temporal Totality:**
S-Space contains the complete tagged history of the observed R-Space segment. Every tag, at every frame, is simultaneously available. You can view frame 1 and frame 1000 at the same instant. You can compare the semantic state at t=5s with t=45s without "waiting" for time to pass.

This is not time travel. This is the difference between reading a book (you can flip to any page) and living a life (you can only move forward). S-Space is the book. R-Space is the life.

**Objects in S-Space are abstract.** They have:
- Position (where in the semantic field)
- Density (how much meaning is concentrated here)
- Entropy (how certain/uncertain the meaning is)
- Velocity (how the meaning is changing over time)
- Relations (how this meaning connects to other meanings)

They do NOT have:
- Mass
- Physical causation
- Material existence
- Unique physical identity (many different horses produce the same tag "horse")

### 1.3 The Abstraction Barrier

Between R-Space and S-Space there is an abstraction barrier. This barrier has asymmetric permeability:

**R-Space → S-Space (Forward Transform / Tagging):**
- MANY-TO-ONE: Many different physical realities can produce the same semantic structure. A brown horse grazing and a white horse grazing both produce the tag "horse_grazing." The physical differences (color, breed, location) are LOST unless explicitly tagged.
- LOSSY: Information is destroyed in the forward transform. The tag "horse" contains less information than the actual horse.
- DETERMINISTIC given a tagger: The same input to the same detection model produces the same tags. (Modulo model stochasticity.)
- IRREVERSIBLE without additional information: You cannot reconstruct the exact horse from the tag "horse."

**S-Space → R-Space (Inverse Transform / Reconstruction):**
- ONE-TO-MANY: A single semantic structure is consistent with infinitely many physical realities. The tags "two_people, profanity, argument, separation, arrest" could describe millions of different actual scenarios.
- GENERATIVE: To go from S-Space back to a specific R-Space instance, you must CHOOSE among the infinite possibilities. This is a creative/generative act, not a deterministic one.
- CONSTRAINED: Not all realities are consistent with a given semantic structure. If the tags say "arrest", no valid reconstruction can omit the arrest. The semantic structure CONSTRAINS possible realities without DETERMINING a specific one.

**CRITICAL LAW 3 — Asymmetric Transform:**
```
Forward:  R → S  is many-to-one (lossy, deterministic, compressive)
Inverse:  S → R  is one-to-many (generative, constrained, expansive)
```

This asymmetry is the foundation of everything THRUPUT does. It means:
1. We can compress reality into meaning (tagging)
2. We can analyze meaning with mathematical tools (engines)
3. We can identify what realities are CONSISTENT with a meaning (constraint analysis)
4. We CANNOT uniquely determine what reality produced a meaning (without the original medium)

## Part II — The Laws of Semantic Space

### 2.1 Conservation Laws

**Law of Semantic Conservation:**
Within a closed S-Space system (one fully tagged segment), the total semantic content is fixed. No meaning is created or destroyed by the mathematical engines — they only reorganize, measure, and expose structure that already exists in the tags.

If the tagger did not tag "horse", no engine can discover a horse. The engines discover RELATIONSHIPS between tags, PATTERNS in how tags distribute over time and space, and STRUCTURE in the semantic field — but they cannot manufacture new content.

**Implication:** The quality of all downstream analysis is bounded by the quality and completeness of tagging. Better taggers = better intelligence. This is why we need rich tagging (not just object labels, but attributes, actions, relations, text content).

### 2.2 The Anchor Principle

**Definition:** An anchor is the tagged object with the highest product of:
- Semantic density (how many tags cluster around it)
- Temporal persistence (how long it persists in the field)
- Relational centrality (how many other objects relate to it)
- Confidence (how certain we are of the detection)

**CRITICAL LAW 4 — Anchor-Reality Equivalence:**
The anchor's tag IS functionally equivalent to the anchor's physical object for all semantic operations. 

Why: Because the anchor has the HIGHEST confidence, LONGEST persistence, and MOST relations, it has been observed so thoroughly that the difference between the tag and the object approaches zero for purposes of semantic analysis. 

An anchor tagged "person_0" with 200 frames of continuous high-confidence detection, with tracked velocity, multiple relation types, and action changes — that tag sequence IS as informationally complete as the original person FOR SEMANTIC PURPOSES.

This does NOT mean the tag IS the person. It means:
- Any semantic operation you perform on the tag yields the same conclusion as if you could perform it on the person directly
- The tag can stand in for the person in any cross-reference, causal analysis, or pattern detection
- The tag-object bond at this level of observation depth is functionally lossless

**For lesser objects** (brief appearances, low confidence), the tag-object bond is lossy and conclusions must be lower confidence.

### 2.3 Semantic Density and Entropy

**Density** at a point in S-Space = how much meaning is concentrated there. High density means many tags, high confidence, persistent objects, rich relations. 

Density is computed from REAL tag data, not fabricated. It is the Gaussian superposition:
```
F(x,y,t) = Σ_i confidence_i × exp(−||position_i − (x,y)||² / 2σ²)
```

**Entropy** at a point in S-Space = how uncertain the meaning is. High entropy means the tags at this point vary a lot over time — the meaning shifts, changes, is unstable. Low entropy means the tags are consistent — the meaning is fixed, reliable, immutable.

**CRITICAL LAW 5 — The Density-Entropy Tradeoff:**
The most valuable points in S-Space are HIGH DENSITY + LOW ENTROPY. These are points where a lot of meaning is concentrated AND that meaning is stable/certain.

These points are the semantic equivalent of primes — irreducible atoms of meaning. You cannot simplify them further. They are the scaffolding around which all other meaning is organized.

### 2.4 How Causation Works in S-Space

In R-Space, causation is physical: ball hits vase, vase breaks. Force → effect.

In S-Space, causation is INFORMATIONAL: the appearance of tag A at time t₁ predicts the appearance of tag B at time t₂ better than tag B's own history predicts it. This is transfer entropy:

```
T(A→B) = H(B_t | B_{t-1}) − H(B_t | B_{t-1}, A_{t-1})
```

In plain English: How much does knowing what A did help me predict what B will do, beyond what B's own past tells me?

**CRITICAL LAW 6 — Semantic Causation ≠ Physical Causation:**
Transfer entropy in S-Space measures INFORMATIONAL influence, not physical force. If "profanity" at t=10 has high transfer entropy toward "arrest" at t=50, it means:
- The profanity event INFORMATIONALLY PRECEDES and PREDICTS the arrest
- The semantic structure of the profanity is connected to the semantic structure of the arrest
- This is consistent with (but does not prove) physical causation

To confirm physical causation, you must:
1. Show informational precedence (transfer entropy)
2. Show temporal ordering (A before B in R-Space time)
3. Show mechanism (plausible causal chain through intermediate tags)
4. Show exclusion (no better alternative explanation)

The program CAN do steps 1-3 from S-Space alone. Step 4 requires either exhaustive tagging or external knowledge.

## Part VI — Glossary of Rigorous Definitions

**Anchor:** The single object in the field with maximum (confidence × persistence × centrality). There is exactly one anchor per analysis. It is the most informationally complete object.

**Tag:** A symbolic label bonded to a specific observation at a specific time. Tags are semantic, not physical. The tag "horse" has no mass.

**Semantic Density F(x,y,t):** The total confidence-weighted tag presence at point (x,y) at time t. Computed as Gaussian superposition. High density = lots of meaning concentrated here.

**Semantic Entropy:** The variance of tag state vectors at a point over time. Low entropy = stable, reliable meaning. High entropy = shifting, uncertain meaning.

**Transfer Entropy T(A→B):** Information-theoretic measure of how much A's past predicts B's future beyond B's own past. Measures informational causation, not physical causation.

**Constraint Tightness:** How narrowly the semantic structure constrains possible realities. Tightness = 1 − (unique_semantic_states / possible_states). High tightness = few realities match this structure. Low tightness = many realities match.

**Semantic Angle:** Cosine of the angle between two objects' state vectors. 0° = identical semantic behavior. 90° = unrelated. 180° = opposite.

**Scaffold Point:** A point in S-Space with high density and low entropy — an irreducible semantic atom. The semantic equivalent of a prime number: it cannot be decomposed further and serves as structural foundation.

**Forward Transform (Tagging):** R-Space → S-Space. Many-to-one. Lossy. Deterministic.

**Inverse Transform (Reconstruction):** S-Space → R-Space. One-to-many. Generative. Constrained.

**Temporal Totality:** The property of S-Space that the entire timeline exists simultaneously as a queryable structure, unlike R-Space where time is experienced sequentially.

**Abstraction Barrier:** The boundary between R-Space and S-Space. Asymmetrically permeable. Forward crossing is compressive. Inverse crossing is expansive.

## Part VII — What We Can Build With This

Given these laws, THRUPUT can:

1. **For any medium:** Extract the complete semantic structure and identify what matters most (anchors)

2. **For text/documents:** Read the content, identify entities and themes, find structural patterns (where the document shifts topic, which entities dominate, what's cross-referenced across distant sections), and produce intelligence about what the content means and implies

3. **For video:** All of the above PLUS spatial-temporal pattern detection, behavioral analysis, causal attribution between objects, prediction of behavioral continuation

4. **For audio:** Spectral pattern analysis, rhythm detection, harmonic structure, speech content analysis (via transcription → text analysis)

5. **Cross-medium:** Take a semantic structure from one medium and express insights about it in another medium (e.g., video analysis expressed as text report)

6. **Causal attribution:** Determine which events in the semantic field most likely caused which other events, using transfer entropy + temporal ordering + propagation matching

7. **Anomaly detection:** Identify points in the semantic field that deviate from the established pattern — things that shouldn't be there, or things that are missing

These are real capabilities. Not theoretical. Not aspirational. These are what the code actually does when it works correctly.
